Okay. I will explain like you are opening Vitis for the first time ever. Don’t worry. Today you will do only this flow:

XSA file → Vitis Platform → Simple print test → MVDR IP test

AMD’s Vitis flow is exactly this: create a platform component from the XSA, then create an application component that runs on the processor and controls the hardware IP. 


---

First understand your setup

Your FPGA board setup is like this:

Linux server / remote PC
        |
        | USB cable
        ↓
FPGA board
        |
        | power adapter
        ↓
Power supply

The USB cable is usually used for:

1. Programming the FPGA through JTAG
2. Seeing printed output through UART/serial terminal

The power plug only powers the board.

PuTTY is just a serial terminal. It shows text printed by your program, like:

Hello World
MVDR IP started
MVDR IP done
w[0] = ...

AMD also says Vitis has a Serial Monitor, and other serial tools like PuTTY can be used.

PART J — First find the MVDR driver name

Before writing full code, we must check what Vitis named your IP driver.

In Vitis, find a file called:

xparameters.h

It may be inside:

mvdr_platform
→ export
→ mvdr_platform
→ sw
→ standalone_psv_cortexa72_0
→ bsp
→ include

or use the search box.

Open xparameters.h.

Press:

Ctrl + F

Search:

MVDR

You need to find something like:

#define XPAR_XMVDR_TOP_AXI_0_DEVICE_ID 0

or something similar.

Also search for the driver header in project files:

xmvdr_top_axi.h

Most likely your include will be:

#include "xmvdr_top_axi.h"

and driver object will be:

XMvdr_top_axi

If your generated name is different, tell me the exact name from xparameters.h.

PART K — MVDR board test code

Paste this into mvdr_ip_test/src/main.c.

This version is made beginner-safe. It starts the IP once, waits with timeout, prints first 8 weights, then stops.

#include "xparameters.h"
#include "xil_cache.h"
#include "xil_printf.h"
#include "xstatus.h"

#include "xmvdr_top_axi.h"

#include <stdint.h>
#include <math.h>

#define NX 8
#define NY 8
#define N  64
#define K  5000
#define SAMPLES (N*K)

#define PI 3.14159265358979323846

static double Xtrain_real[SAMPLES] __attribute__((aligned(64)));
static double Xtrain_imag[SAMPLES] __attribute__((aligned(64)));

static double w_real[N] __attribute__((aligned(64)));
static double w_imag[N] __attribute__((aligned(64)));

static double a_s_real[N];
static double a_s_imag[N];

static double a_i1_real[N];
static double a_i1_imag[N];

static double a_i2_real[N];
static double a_i2_imag[N];

static uint32_t rng_state = 123456789u;

static double rand_unit(void)
{
    rng_state = 1664525u * rng_state + 1013904223u;
    uint32_t v = (rng_state >> 8) & 0x00FFFFFFu;
    return ((double)v / 16777215.0) * 2.0 - 1.0;
}

static void make_steering(double az_deg, double el_deg, double d,
                          double ar[N], double ai[N])
{
    double az = az_deg * PI / 180.0;
    double el = el_deg * PI / 180.0;

    double u = sin(az) * cos(el);
    double v = sin(el);

    int index = 0;

    for (int y = 0; y < NY; y++) {
        for (int x = 0; x < NX; x++) {
            double phase = 2.0 * PI * d * ((double)x * u + (double)y * v);

            ar[index] = cos(phase) / sqrt((double)N);
            ai[index] = sin(phase) / sqrt((double)N);

            index++;
        }
    }
}

static void generate_training_data(void)
{
    double d = 0.5;

    make_steering(40.0,   0.0,  d, a_s_real,  a_s_imag);
    make_steering(30.0,  10.0,  d, a_i1_real, a_i1_imag);
    make_steering(-60.0, -25.0, d, a_i2_real, a_i2_imag);

    for (int k = 0; k < K; k++) {
        double t = (double)k / (double)K;

        double sr = cos(2.0 * PI * 50.0 * t);
        double si = sin(2.0 * PI * 50.0 * t);

        double i1r = rand_unit();
        double i1i = rand_unit();

        double i2r = rand_unit();
        double i2i = rand_unit();

        for (int n = 0; n < N; n++) {
            double xr = 0.0;
            double xi = 0.0;

            xr += 10.0 * (sr * a_s_real[n] - si * a_s_imag[n]);
            xi += 10.0 * (sr * a_s_imag[n] + si * a_s_real[n]);

            xr += 5.0 * (i1r * a_i1_real[n] - i1i * a_i1_imag[n]);
            xi += 5.0 * (i1r * a_i1_imag[n] + i1i * a_i1_real[n]);

            xr += 5.0 * (i2r * a_i2_real[n] - i2i * a_i2_imag[n]);
            xi += 5.0 * (i2r * a_i2_imag[n] + i2i * a_i2_real[n]);

            xr += 0.1 * rand_unit();
            xi += 0.1 * rand_unit();

            int idx = n * K + k;

            Xtrain_real[idx] = xr;
            Xtrain_imag[idx] = xi;
        }
    }

    for (int n = 0; n < N; n++) {
        w_real[n] = 0.0;
        w_imag[n] = 0.0;
    }
}

static void print_double_6(double value)
{
    int neg = 0;

    if (value < 0.0) {
        neg = 1;
        value = -value;
    }

    int scaled = (int)(value * 1000000.0 + 0.5);
    int whole = scaled / 1000000;
    int frac  = scaled % 1000000;

    if (neg) {
        xil_printf("-");
    }

    xil_printf("%d.%06d", whole, frac);
}

int main(void)
{
    xil_printf("\r\n==============================\r\n");
    xil_printf("MVDR AXI IP Board Test Started\r\n");
    xil_printf("==============================\r\n");

    xil_printf("Generating training data...\r\n");
    generate_training_data();
    xil_printf("Training data generated.\r\n");

    xil_printf("First input sample = ");
    print_double_6(Xtrain_real[0]);
    xil_printf(" + j");
    print_double_6(Xtrain_imag[0]);
    xil_printf("\r\n");

    XMvdr_top_axi ip;

    int status = XMvdr_top_axi_Initialize(&ip, XPAR_XMVDR_TOP_AXI_0_DEVICE_ID);

    if (status != XST_SUCCESS) {
        xil_printf("ERROR: MVDR IP initialization failed.\r\n");
        return -1;
    }

    xil_printf("MVDR IP initialized.\r\n");

    Xil_DCacheFlushRange((UINTPTR)Xtrain_real, sizeof(Xtrain_real));
    Xil_DCacheFlushRange((UINTPTR)Xtrain_imag, sizeof(Xtrain_imag));
    Xil_DCacheFlushRange((UINTPTR)w_real, sizeof(w_real));
    Xil_DCacheFlushRange((UINTPTR)w_imag, sizeof(w_imag));

    XMvdr_top_axi_Set_Xtrain_real(&ip, (UINTPTR)Xtrain_real);
    XMvdr_top_axi_Set_Xtrain_imag(&ip, (UINTPTR)Xtrain_imag);

    XMvdr_top_axi_Set_w_real(&ip, (UINTPTR)w_real);
    XMvdr_top_axi_Set_w_imag(&ip, (UINTPTR)w_imag);

    XMvdr_top_axi_Set_az_deg(&ip, 40.0);
    XMvdr_top_axi_Set_el_deg(&ip, 0.0);
    XMvdr_top_axi_Set_d(&ip, 0.5);

    xil_printf("Starting MVDR IP...\r\n");

    XMvdr_top_axi_Start(&ip);

    volatile unsigned int timeout = 0xFFFFFFFFu;

    while (!XMvdr_top_axi_IsDone(&ip) && timeout > 0u) {
        timeout--;
    }

    if (timeout == 0u) {
        xil_printf("ERROR: MVDR IP timeout. ap_done did not arrive.\r\n");
        xil_printf("Stop this run. Do not keep waiting forever.\r\n");
        return -2;
    }

    xil_printf("MVDR IP done.\r\n");

    Xil_DCacheInvalidateRange((UINTPTR)w_real, sizeof(w_real));
    Xil_DCacheInvalidateRange((UINTPTR)w_imag, sizeof(w_imag));

    xil_printf("\r\nFirst 8 MVDR weights:\r\n");

    for (int i = 0; i < 8; i++) {
        xil_printf("w[%d] = ", i);
        print_double_6(w_real[i]);
        xil_printf(" + j");
        print_double_6(w_imag[i]);
        xil_printf("\r\n");
    }

    xil_printf("\r\nMVDR AXI IP Board Test Completed Successfully.\r\n");

    while (1) {
        // Stop here after printing
    }

    return 0;
}

PART L — If math error comes

If build gives error like:

undefined reference to sin
undefined reference to cos
undefined reference to sqrt

you must add math library.

Try this:

1. Right-click mvdr_ip_test.


2. Click Settings or C/C++ Build Settings.


3. Find Libraries.


4. Add:



m

Not -lm, just m if it asks library name.

If there is a place called Other linker flags, add:

-lm

Then build again.


---

PART M — If memory overflow error comes

If build says something like:

section .bss will not fit
region overflowed

then your big arrays are going into small memory. You need linker script to use DDR.

Find file:

lscript.ld

Open it.

Look for memory region. You need large sections like .bss, .data, heap, stack to go to DDR, not OCM.

If you get this error, send me screenshot. Don’t guess.

How long to wait

For Hello World:

If no output in 2 minutes, stop and check serial/JTAG.

For MVDR:

If no “MVDR IP done” within 5–10 minutes, stop.

Do not leave it running for 30 minutes.

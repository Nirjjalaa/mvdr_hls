backsub_top.cpp

This is your verified algorithm with HLS pragmas.

#include "backsub_top.h"

void backsub_top(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
#pragma HLS INTERFACE m_axi port=R offset=slave bundle=gmem0
#pragma HLS INTERFACE m_axi port=b offset=slave bundle=gmem1
#pragma HLS INTERFACE m_axi port=x offset=slave bundle=gmem2
#pragma HLS INTERFACE s_axilite port=return

    for(int i=N-1;i>=0;i--)
    {
        x[i] = b[i];

        for(int j=i+1;j<N;j++)
        {
#pragma HLS PIPELINE
            x[i] -= R[i][j]*x[j];
        }

        x[i] /= R[i][i];
    }
}


---

Testbench for 64×64

The trick is:

We choose a known solution:

x_true = [1,1,1,...,1]

and construct:

b = R*x_true

Then the answer must be exactly:

x = [1,1,1,...,1]

tb_backsub.cpp

#include <iostream>
#include <cmath>

#include "backsub_top.h"

using namespace std;

int main()
{
    static cplx R[N][N];
    static cplx b[N];
    static cplx x[N];

    //---------------------------------
    // Build upper triangular matrix
    //---------------------------------

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            if(i==j)
                R[i][j]=cplx(2.0,0.0);

            else if(j>i)
                R[i][j]=cplx(1.0,0.0);

            else
                R[i][j]=cplx(0.0,0.0);
        }
    }

    //---------------------------------
    // Build b=R*xtrue
    //---------------------------------

    for(int i=0;i<N;i++)
    {
        double sum=2.0;

        for(int j=i+1;j<N;j++)
            sum+=1.0;

        b[i]=cplx(sum,0.0);
    }

    //---------------------------------
    // Call DUT
    //---------------------------------

    backsub_top(R,b,x);

    //---------------------------------
    // Verify
    //---------------------------------

    double max_error=0;

    cout<<"\nFirst 10 outputs:\n"<<endl;

    for(int i=0;i<10;i++)
    {
        double err=
            abs(x[i]-cplx(1.0,0.0));

        if(err>max_error)
            max_error=err;

        cout
            <<"x["<<i<<"] = "
            <<x[i].real()
            <<" + "
            <<x[i].imag()
            <<"j"<<endl;
    }

    for(int i=0;i<N;i++)
    {
        double err=
            abs(x[i]-cplx(1.0,0.0));

        if(err>max_error)
            max_error=err;
    }

    cout<<"\nMaximum error = "
        <<max_error
        <<endl;

    if(max_error<1e-9)
    {
        cout<<"\nTEST PASSED"<<endl;
        return 0;
    }
    else
    {
        cout<<"\nTEST FAILED"<<endl;
        return 1;
    }
}

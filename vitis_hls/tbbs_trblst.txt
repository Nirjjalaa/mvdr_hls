#include <iostream>
#include <cmath>
#include "backsub_top.h"

using namespace std;

int main()
{
    static cplx R[N][N];
    static cplx b[N];
    static cplx x[N];

    // Build upper triangular matrix
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            if(i==j)
                R[i][j]=cplx(2.0,0.0);
            else if(j>i)
                R[i][j]=cplx(1.0,0.0);
            else
                R[i][j]=cplx(0.0,0.0);
        }
    }

    // Build b = R*x_true
    // where x_true = [1,1,1,...]
    for(int i=0;i<N;i++)
    {
        double sum=2.0;

        for(int j=i+1;j<N;j++)
            sum += 1.0;

        b[i]=cplx(sum,0.0);
    }

    // Run DUT
    backsub_top(R,b,x);

    // Verify
    double max_error=0;

    cout << "\nFirst 10 outputs:\n";

    for(int i=0;i<10;i++)
    {
        double err = abs(x[i]-cplx(1.0,0.0));

        if(err>max_error)
            max_error=err;

        cout << "x[" << i << "] = "
             << x[i].real()
             << " + "
             << x[i].imag()
             << "j\n";
    }

    for(int i=0;i<N;i++)
    {
        double err=abs(x[i]-cplx(1.0,0.0));

        if(err>max_error)
            max_error=err;
    }

    cout << "\nMaximum error = "
         << max_error << endl;

    if(max_error<1e-9)
    {
        cout << "\nTEST PASSED\n";
        return 0;
    }
    else
    {
        cout << "\nTEST FAILED\n";
        return 1;
    }
}

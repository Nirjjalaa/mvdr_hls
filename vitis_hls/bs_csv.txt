void backsub_top(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
#pragma HLS INTERFACE m_axi port=R offset=slave bundle=gmem0
#pragma HLS INTERFACE m_axi port=b offset=slave bundle=gmem1
#pragma HLS INTERFACE m_axi port=x offset=slave bundle=gmem2
#pragma HLS INTERFACE s_axilite port=return

    for(int i=N-1;i>=0;i--)
    {
        x[i]=b[i];

        for(int j=i+1;j<N;j++)
        {
#pragma HLS PIPELINE
            x[i]-=R[i][j]*x[j];
        }

        x[i]/=R[i][i];
    }
}

#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <cmath>

#include "backsub_top.h"

using namespace std;

//=====================================================
// Load 1D CSV
//=====================================================
void load_1d(
    const string& file,
    float data[],
    int n)
{
    ifstream fin(file);

    if(!fin.is_open())
    {
        cout << "Cannot open "
             << file << endl;
        exit(1);
    }

    string line;
    int i=0;

    while(getline(fin,line) && i<n)
    {
        if(line.empty())
            continue;

        data[i]=stof(line);
        i++;
    }

    fin.close();
}

//=====================================================
// Load 2D CSV
//=====================================================
void load_2d(
    const string& file,
    float data[][N],
    int rows)
{
    ifstream fin(file);

    if(!fin.is_open())
    {
        cout << "Cannot open "
             << file << endl;
        exit(1);
    }

    string line;
    int i=0;

    while(getline(fin,line) && i<rows)
    {
        if(line.empty())
            continue;

        stringstream ss(line);

        string value;
        int j=0;

        while(getline(ss,value,','))
        {
            data[i][j]=stof(value);
            j++;
        }

        i++;
    }

    fin.close();
}

//=====================================================
int main()
{
    cout << "\n";
    cout << "====================================\n";
    cout << "MVDR BACK SUBSTITUTION HLS TESTBENCH\n";
    cout << "====================================\n";

    //-------------------------------------------------
    // Storage
    //-------------------------------------------------

    static float Ru_re[N][N];
    static float Ru_im[N][N];

    static float y_re[N];
    static float y_im[N];

    static float w_re[N];
    static float w_im[N];

    static float as_re[N];
    static float as_im[N];

    //-------------------------------------------------
    // Load CSV files
    //-------------------------------------------------

    cout << "Loading CSV files...\n";

    load_2d(
        CSV_PATH "Ru_real.csv",
        Ru_re,
        N);

    load_2d(
        CSV_PATH "Ru_imag.csv",
        Ru_im,
        N);

    load_1d(
        CSV_PATH "y_real.csv",
        y_re,
        N);

    load_1d(
        CSV_PATH "y_imag.csv",
        y_im,
        N);

    load_1d(
        CSV_PATH "w_real.csv",
        w_re,
        N);

    load_1d(
        CSV_PATH "w_imag.csv",
        w_im,
        N);

    load_1d(
        CSV_PATH "a_s_real.csv",
        as_re,
        N);

    load_1d(
        CSV_PATH "a_s_imag.csv",
        as_im,
        N);

    cout << "CSV loading complete.\n";

    //-------------------------------------------------
    // Construct complex arrays
    //-------------------------------------------------

    static cplx R[N][N];
    static cplx y[N];
    static cplx x[N];
    static cplx w_matlab[N];
    static cplx a_s[N];

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            R[i][j]=cplx(
                Ru_re[i][j],
                Ru_im[i][j]);
        }

        y[i]=cplx(
            y_re[i],
            y_im[i]);

        w_matlab[i]=cplx(
            w_re[i],
            w_im[i]);

        a_s[i]=cplx(
            as_re[i],
            as_im[i]);
    }

    //-------------------------------------------------
    // Run HLS DUT
    //-------------------------------------------------

    cout << "\nRunning backsub_top()...\n";

    backsub_top(
        R,
        y,
        x);

    cout << "Done.\n";

    //-------------------------------------------------
    // Normalize
    //-------------------------------------------------

    cplx norm_factor(0,0);

    for(int i=0;i<N;i++)
        norm_factor +=
            conj(a_s[i])*x[i];

    for(int i=0;i<N;i++)
        x[i] /= norm_factor;

    //-------------------------------------------------
    // Print first 5 weights
    //-------------------------------------------------

    cout << "\nFirst 5 weights:\n";

    for(int i=0;i<5;i++)
    {
        cout
        << i
        << " : MATLAB = "
        << w_matlab[i]
        << "    HLS = "
        << x[i]
        << endl;
    }

    //-------------------------------------------------
    // Relative error
    //-------------------------------------------------

    double err=0.0;
    double normw=0.0;

    for(int i=0;i<N;i++)
    {
        err += norm(
            x[i]-w_matlab[i]);

        normw += norm(
            w_matlab[i]);
    }

    double rel_err=
        sqrt(err/normw);

    //-------------------------------------------------
    // Final result
    //-------------------------------------------------

    cout << "\n";
    cout << "Relative Error = "
         << rel_err
         << endl;

    if(rel_err < 1e-3)
    {
        cout
        << "\nBACKSUB HLS : PASS\n";

        return 0;
    }
    else
    {
        cout
        << "\nBACKSUB HLS : FAIL\n";

        return 1;
    }
}

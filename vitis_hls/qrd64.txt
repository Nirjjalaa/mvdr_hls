Step 2: Create CSV loading functions

Paste this at the top of tb_qrd.cpp.

#include <iostream>
#include <fstream>
#include <string>
#include <cmath>

#include "qrd_top.h"

using namespace std;

void load_csv(
    const string& filename,
    float data[][N])
{
    ifstream file(filename);

    string value;

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            getline(file,value,',');

            if(j==N-1)
            {
                size_t pos=value.find('\n');
                if(pos!=string::npos)
                    value=value.substr(0,pos);
            }

            data[i][j]=stof(value);
        }
    }

    file.close();
}

Step 3: Relative error function

Paste below it.

float matrix_error(
    cplx A[N][N],
    cplx B[N][N])
{
    double err=0;
    double normA=0;

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            err+=norm(A[i][j]-B[i][j]);
            normA+=norm(A[i][j]);
        }
    }

    return sqrt(err)/sqrt(normA);
}

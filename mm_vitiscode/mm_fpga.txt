Very important: do not export Xtrain(:).

This is correct:

writematrix(real(Xtrain), 'Xtrain_real.csv');

This is wrong for our FPGA order:

writematrix(real(Xtrain(:)), 'Xtrain_real.csv');

Why? Because our HLS/FPGA expects:

index = sensor_index * K + snapshot_index

That means:

Xtrain_data[0]    = MATLAB Xtrain(1,1)
Xtrain_data[1]    = MATLAB Xtrain(1,2)
Xtrain_data[5000] = MATLAB Xtrain(2,1)

import re

N = 64
K = 5000
SAMPLES = N * K

def read_rows(filename):
    rows = []

    with open(filename, "r", encoding="utf-8") as f:
        for line in f:
            tokens = re.split(r"[,\s]+", line.strip())
            values = [float(t) for t in tokens if t != ""]
            if values:
                rows.append(values)

    return rows

def flatten_xtrain_sensor_major(filename):
    rows = read_rows(filename)

    # Correct case:
    # MATLAB exported 64 rows and 5000 columns using writematrix(real(Xtrain))
    if len(rows) == N and all(len(row) == K for row in rows):
        flat = []

        for sensor in range(N):
            for snapshot in range(K):
                flat.append(rows[sensor][snapshot])

        return flat

    # Wrong case:
    # MATLAB exported one long vector
    all_values = []

    for row in rows:
        all_values.extend(row)

    if len(all_values) == SAMPLES:
        raise ValueError(
            filename + " contains 320000 values but not as 64 x 5000 matrix.\n"
            "Regenerate it in MATLAB using:\n"
            "writematrix(real(Xtrain), 'Xtrain_real.csv')\n"
            "Do not use Xtrain(:)."
        )

    raise ValueError(
        filename + " has wrong shape.\n"
        "Expected 64 rows and 5000 columns."
    )

def read_vector_64(filename):
    rows = read_rows(filename)

    values = []

    for row in rows:
        values.extend(row)

    if len(values) != N:
        raise ValueError(
            filename + " should contain exactly 64 values."
        )

    return values

def write_array(f, ctype, name, values):
    f.write(f"static const {ctype} {name}[{len(values)}] __attribute__((aligned(64))) = {{\n")

    for i, v in enumerate(values):
        f.write(f"{v:.17g}")

        if i != len(values) - 1:
            f.write(",")

        if (i + 1) % 4 == 0:
            f.write("\n")
        else:
            f.write(" ")

    f.write("\n};\n\n")

print("Reading MATLAB CSV files...")

xreal = flatten_xtrain_sensor_major("Xtrain_real.csv")
ximag = flatten_xtrain_sensor_major("Xtrain_imag.csv")
wreal = read_vector_64("w_real.csv")
wimag = read_vector_64("w_imag.csv")

print("Lengths:")
print("Xtrain_real:", len(xreal))
print("Xtrain_imag:", len(ximag))
print("w_real:", len(wreal))
print("w_imag:", len(wimag))

print("\nOrder sanity check:")
print("Xtrain_data[0]    = MATLAB Xtrain(1,1) =", xreal[0])
print("Xtrain_data[1]    = MATLAB Xtrain(1,2) =", xreal[1])
print("Xtrain_data[5000] = MATLAB Xtrain(2,1) =", xreal[5000])

print("\nFirst 8 golden MATLAB weights:")
for i in range(8):
    print(f"w[{i}] = {wreal[i]} + j{wimag[i]}")

with open("mvdr_data.h", "w", encoding="utf-8") as f:
    f.write("#ifndef MVDR_DATA_H\n")
    f.write("#define MVDR_DATA_H\n\n")

    f.write("#define N 64\n")
    f.write("#define K 5000\n")
    f.write("#define SAMPLES 320000\n\n")

    write_array(f, "double", "Xtrain_real_data", xreal)
    write_array(f, "double", "Xtrain_imag_data", ximag)
    write_array(f, "double", "golden_w_real", wreal)
    write_array(f, "double", "golden_w_imag", wimag)

    f.write("#endif\n")

print("\nDone. Created mvdr_data.h")

PART 8 — Paste complete main.c

Open main.c.

Paste this full code.

This code includes the changes you already discovered:

UINTPTR changed to (uint64_t)(uintptr_t)
double scalar register values passed as 64-bit double bits
asm volatile("nop") in timeout loop

#include "xparameters.h"
#include "xil_cache.h"
#include "xil_printf.h"
#include "xstatus.h"

#include "xmvdr_top_axi.h"

#include <stdint.h>
#include <stdint.h>

#include "mvdr_data.h"

static double fpga_w_real[N] __attribute__((aligned(64)));
static double fpga_w_imag[N] __attribute__((aligned(64)));

static uint64_t double_to_u64(double x)
{
    union {
        double d;
        uint64_t u;
    } converter;

    converter.d = x;
    return converter.u;
}

static double abs_double(double x)
{
    if (x < 0.0) {
        return -x;
    }

    return x;
}

static void print_double_6(double value)
{
    int neg = 0;

    if (value < 0.0) {
        neg = 1;
        value = -value;
    }

    int scaled = (int)(value * 1000000.0 + 0.5);
    int whole = scaled / 1000000;
    int frac = scaled % 1000000;

    if (neg) {
        xil_printf("-");
    }

    xil_printf("%d.%06d", whole, frac);
}

static void print_double_9_small(double value)
{
    int neg = 0;

    if (value < 0.0) {
        neg = 1;
        value = -value;
    }

    /*
     * This function is for small error values.
     * It prints 9 digits after decimal.
     */
    int scaled = (int)(value * 1000000000.0 + 0.5);
    int whole = scaled / 1000000000;
    int frac = scaled % 1000000000;

    if (neg) {
        xil_printf("-");
    }

    xil_printf("%d.%09d", whole, frac);
}

int main(void)
{
    xil_printf("\r\n");
    xil_printf("========================================\r\n");
    xil_printf("MVDR FPGA vs MATLAB Verification Started\r\n");
    xil_printf("========================================\r\n");

    xil_printf("N       = %d\r\n", N);
    xil_printf("K       = %d\r\n", K);
    xil_printf("Samples = %d\r\n", SAMPLES);

    /*
     * Clear FPGA output arrays before starting IP.
     */
    for (int i = 0; i < N; i++) {
        fpga_w_real[i] = 0.0;
        fpga_w_imag[i] = 0.0;
    }

    xil_printf("\r\nChecking MATLAB data order:\r\n");

    xil_printf("Xtrain_data[0]    should be MATLAB Xtrain(1,1) = ");
    print_double_6(Xtrain_real_data[0]);
    xil_printf(" + j");
    print_double_6(Xtrain_imag_data[0]);
    xil_printf("\r\n");

    xil_printf("Xtrain_data[1]    should be MATLAB Xtrain(1,2) = ");
    print_double_6(Xtrain_real_data[1]);
    xil_printf(" + j");
    print_double_6(Xtrain_imag_data[1]);
    xil_printf("\r\n");

    xil_printf("Xtrain_data[5000] should be MATLAB Xtrain(2,1) = ");
    print_double_6(Xtrain_real_data[5000]);
    xil_printf(" + j");
    print_double_6(Xtrain_imag_data[5000]);
    xil_printf("\r\n");

    xil_printf("\r\nFirst 8 MATLAB golden weights from header:\r\n");

    for (int i = 0; i < 8; i++) {
        xil_printf("MATLAB w[%d] = ", i);
        print_double_6(golden_w_real[i]);
        xil_printf(" + j");
        print_double_6(golden_w_imag[i]);
        xil_printf("\r\n");
    }

    XMvdr_top_axi ip;

    xil_printf("\r\nInitializing MVDR IP...\r\n");

    int status = XMvdr_top_axi_Initialize(&ip, XPAR_XMVDR_TOP_AXI_0_DEVICE_ID);

    if (status != XST_SUCCESS) {
        xil_printf("ERROR: MVDR IP initialization failed.\r\n");
        return -1;
    }

    xil_printf("MVDR IP initialized successfully.\r\n");

    /*
     * Cache flush before hardware starts.
     * This makes sure the FPGA reads updated data from DDR.
     */
    xil_printf("Flushing cache...\r\n");

    Xil_DCacheFlushRange((uintptr_t)Xtrain_real_data, sizeof(double) * SAMPLES);
    Xil_DCacheFlushRange((uintptr_t)Xtrain_imag_data, sizeof(double) * SAMPLES);
    Xil_DCacheFlushRange((uintptr_t)fpga_w_real, sizeof(fpga_w_real));
    Xil_DCacheFlushRange((uintptr_t)fpga_w_imag, sizeof(fpga_w_imag));

    /*
     * Pointer arguments:
     * Use 64-bit addresses on Versal.
     */
    xil_printf("Setting AXI pointer registers...\r\n");

    XMvdr_top_axi_Set_Xtrain_real(&ip, (uint64_t)(uintptr_t)Xtrain_real_data);
    XMvdr_top_axi_Set_Xtrain_imag(&ip, (uint64_t)(uintptr_t)Xtrain_imag_data);
    XMvdr_top_axi_Set_w_real(&ip,      (uint64_t)(uintptr_t)fpga_w_real);
    XMvdr_top_axi_Set_w_imag(&ip,      (uint64_t)(uintptr_t)fpga_w_imag);

    /*
     * Scalar arguments:
     * az_deg, el_deg and d are double in HLS.
     * So we send the exact 64-bit binary representation of double.
     */
    double az_val = 40.0;
    double el_val = 0.0;
    double d_val  = 0.5;

    xil_printf("Setting scalar registers...\r\n");

    XMvdr_top_axi_Set_az_deg(&ip, double_to_u64(az_val));
    XMvdr_top_axi_Set_el_deg(&ip, double_to_u64(el_val));
    XMvdr_top_axi_Set_d(&ip,      double_to_u64(d_val));

    xil_printf("\r\nStarting MVDR FPGA IP...\r\n");

    XMvdr_top_axi_Start(&ip);

    /*
     * Timeout wait.
     * Do not wait forever.
     */
    volatile unsigned int timeout = 0xFFFFFFFFu;

    while (!XMvdr_top_axi_IsDone(&ip) && timeout > 0u) {
        asm volatile("nop");
        timeout--;
    }

    if (timeout == 0u) {
        xil_printf("ERROR: MVDR IP timeout. ap_done did not arrive.\r\n");
        xil_printf("Do not keep running. Stop and check AXI/DDR/cache/registers.\r\n");
        return -2;
    }

    xil_printf("MVDR FPGA IP done.\r\n");

    /*
     * Cache invalidate after hardware finishes.
     * This makes CPU read FPGA-written output from DDR.
     */
    xil_printf("Invalidating output cache...\r\n");

    Xil_DCacheInvalidateRange((uintptr_t)fpga_w_real, sizeof(fpga_w_real));
    Xil_DCacheInvalidateRange((uintptr_t)fpga_w_imag, sizeof(fpga_w_imag));

    xil_printf("\r\nFirst 8 FPGA weights:\r\n");

    for (int i = 0; i < 8; i++) {
        xil_printf("FPGA  w[%d] = ", i);
        print_double_6(fpga_w_real[i]);
        xil_printf(" + j");
        print_double_6(fpga_w_imag[i]);
        xil_printf("\r\n");
    }

    xil_printf("\r\nFirst 8 MATLAB golden weights:\r\n");

    for (int i = 0; i < 8; i++) {
        xil_printf("MATLAB w[%d] = ", i);
        print_double_6(golden_w_real[i]);
        xil_printf(" + j");
        print_double_6(golden_w_imag[i]);
        xil_printf("\r\n");
    }

    /*
     * Numerical error calculation.
     */
    double max_real_error = 0.0;
    double max_imag_error = 0.0;
    double max_component_error = 0.0;

    double sum_error_sq = 0.0;
    double sum_golden_sq = 0.0;

    for (int i = 0; i < N; i++) {
        double er = fpga_w_real[i] - golden_w_real[i];
        double ei = fpga_w_imag[i] - golden_w_imag[i];

        double abs_er = abs_double(er);
        double abs_ei = abs_double(ei);

        if (abs_er > max_real_error) {
            max_real_error = abs_er;
        }

        if (abs_ei > max_imag_error) {
            max_imag_error = abs_ei;
        }

        if (abs_er > max_component_error) {
            max_component_error = abs_er;
        }

        if (abs_ei > max_component_error) {
            max_component_error = abs_ei;
        }

        sum_error_sq += er * er + ei * ei;
        sum_golden_sq += golden_w_real[i] * golden_w_real[i]
                       + golden_w_imag[i] * golden_w_imag[i];
    }

    xil_printf("\r\nVerification Results:\r\n");

    xil_printf("Max real error      = ");
    print_double_9_small(max_real_error);
    xil_printf("\r\n");

    xil_printf("Max imaginary error = ");
    print_double_9_small(max_imag_error);
    xil_printf("\r\n");

    xil_printf("Max component error = ");
    print_double_9_small(max_component_error);
    xil_printf("\r\n");

    xil_printf("Error power         = ");
    print_double_9_small(sum_error_sq);
    xil_printf("\r\n");

    xil_printf("Golden weight power = ");
    print_double_9_small(sum_golden_sq);
    xil_printf("\r\n");

    if (max_component_error < 0.000001) {
        xil_printf("\r\nRESULT: FPGA output matches MATLAB golden output very closely.\r\n");
    } else if (max_component_error < 0.001) {
        xil_printf("\r\nRESULT: FPGA output matches MATLAB with small numerical error.\r\n");
    } else {
        xil_printf("\r\nRESULT: Error is larger. Check input ordering or data mismatch.\r\n");
    }

    xil_printf("\r\nPrinting all 64 FPGA weights:\r\n");

    for (int i = 0; i < N; i++) {
        xil_printf("FPGA w[%d] = ", i);
        print_double_6(fpga_w_real[i]);
        xil_printf(" + j");
        print_double_6(fpga_w_imag[i]);
        xil_printf("\r\n");
    }

    xil_printf("\r\nMVDR FPGA vs MATLAB Verification Completed.\r\n");

    while (1) {
        asm volatile("nop");
    }

    return 0;
}

steering_vec.cpp

#include "steering_vec.h"
#include <cmath>

void steering_vec(
    double az_deg,
    double el_deg,
    double d,
    cplx a[N]
)
{
    const double DEG_TO_RAD =
        PI / 180.0;

    const double NORMALIZATION =
        0.125;  // 1/sqrt(64)

    double az =
        az_deg * DEG_TO_RAD;

    double el =
        el_deg * DEG_TO_RAD;

    double u =
        std::sin(az) *
        std::cos(el);

    double v =
        std::sin(el);

STEERING_LOOP:
    for (int index = 0; index < N; index++)
    {
#pragma HLS PIPELINE

        int x = index % NX;
        int y = index / NX;

        double phase =
            2.0 * PI * d *
            (
                (double)x * u +
                (double)y * v
            );

        double real_value =
            NORMALIZATION *
            std::cos(phase);

        double imag_value =
            NORMALIZATION *
            std::sin(phase);

        a[index] =
            cplx(real_value, imag_value);
    }
}

This produces the same complex steering vector:

real = 0.125 × cos(phase)
imag = 0.125 × sin(phase)

2. Final covariance source

This version computes only one triangle and creates the other triangle using Hermitian symmetry:

R_{j,i}=R_{i,j}^{*}

That almost halves the number of covariance calculations.

Keep covariance.h and tb_covariance.cpp. Replace only covariance.cpp.

covariance.cpp

#include "covariance.h"

void covariance_matrix(
    const cplx Xtrain[N][K],
    cplx R[N][N]
)
{
    const double INV_K =
        1.0 / (double)K;

    /*
     * Compute the upper triangle, including
     * the main diagonal.
     *
     * R(i,j) =
     * sum X(i,k) * conj(X(j,k)) / K
     */
COVARIANCE_ROW:
    for (int i = 0; i < N; i++)
    {
    COVARIANCE_COLUMN:
        for (int j = i; j < N; j++)
        {
            double sum_real = 0.0;
            double sum_imag = 0.0;

        SNAPSHOT_LOOP:
            for (int k = 0; k < K; k++)
            {
#pragma HLS PIPELINE

                double xi_real =
                    Xtrain[i][k].real();

                double xi_imag =
                    Xtrain[i][k].imag();

                double xj_real =
                    Xtrain[j][k].real();

                double xj_imag =
                    Xtrain[j][k].imag();

                /*
                 * X(i,k) * conj(X(j,k))
                 */
                sum_real +=
                    xi_real * xj_real +
                    xi_imag * xj_imag;

                sum_imag +=
                    xi_imag * xj_real -
                    xi_real * xj_imag;
            }

            double covariance_real =
                sum_real * INV_K;

            double covariance_imag =
                sum_imag * INV_K;

            R[i][j] =
                cplx(
                    covariance_real,
                    covariance_imag
                );

            /*
             * Hermitian mirror:
             *
             * R(j,i) = conj(R(i,j))
             */
            if (i != j)
            {
                R[j][i] =
                    cplx(
                        covariance_real,
                        -covariance_imag
                    );
            }
        }
    }

    /*
     * Diagonal loading:
     *
     * delta = 0.01 * trace(R) / N
     */
    double trace_real = 0.0;

TRACE_LOOP:
    for (int i = 0; i < N; i++)
    {
#pragma HLS PIPELINE

        trace_real +=
            R[i][i].real();
    }

    double delta =
        0.01 *
        trace_real /
        (double)N;

DIAGONAL_LOADING_LOOP:
    for (int i = 0; i < N; i++)
    {
#pragma HLS PIPELINE

        R[i][i] =
            cplx(
                R[i][i].real() + delta,
                0.0
            );
    }
}

The snapshot and trace accumulation loops may still report a dependency-based II warning. That is expected because each accumulated sum depends on its preceding value.

3. Final complex QRD source

Keep qr.h and tb_qr.cpp. Replace only qr.cpp.

qr.cpp

#include "qr.h"
#include <cmath>

static double complex_mag_squared(
    cplx value
)
{
#pragma HLS INLINE

    double real_part =
        value.real();

    double imag_part =
        value.imag();

    return
        real_part * real_part +
        imag_part * imag_part;
}

static double complex_magnitude(
    cplx value
)
{
#pragma HLS INLINE

    return std::sqrt(
        complex_mag_squared(value)
    );
}

static cplx complex_conjugate(
    cplx value
)
{
#pragma HLS INLINE

    return cplx(
        value.real(),
        -value.imag()
    );
}

static cplx complex_scale(
    cplx value,
    double scale
)
{
#pragma HLS INLINE

    return cplx(
        value.real() * scale,
        value.imag() * scale
    );
}

static cplx complex_add(
    cplx first,
    cplx second
)
{
#pragma HLS INLINE

    return cplx(
        first.real() + second.real(),
        first.imag() + second.imag()
    );
}

static cplx complex_subtract(
    cplx first,
    cplx second
)
{
#pragma HLS INLINE

    return cplx(
        first.real() - second.real(),
        first.imag() - second.imag()
    );
}

static cplx complex_multiply(
    cplx first,
    cplx second
)
{
#pragma HLS INLINE

    double ar = first.real();
    double ai = first.imag();

    double br = second.real();
    double bi = second.imag();

    return cplx(
        ar * br - ai * bi,
        ar * bi + ai * br
    );
}

static cplx complex_divide_real(
    cplx numerator,
    double denominator
)
{
#pragma HLS INLINE

    if (
        denominator > -1.0e-30 &&
        denominator < 1.0e-30
    )
    {
        return cplx(0.0, 0.0);
    }

    return cplx(
        numerator.real() / denominator,
        numerator.imag() / denominator
    );
}

void givens_qr(
    cplx A[N][N],
    cplx Q[N][N],
    cplx R[N][N]
)
{
    const double EPS =
        1.0e-12;

    const double EPS_SQUARED =
        1.0e-24;

    /*
     * R = A
     * Q = identity
     */
INITIALIZE_ROWS:
    for (int row = 0; row < N; row++)
    {
    INITIALIZE_COLUMNS:
        for (int column = 0;
             column < N;
             column++)
        {
#pragma HLS PIPELINE

            R[row][column] =
                A[row][column];

            if (row == column)
            {
                Q[row][column] =
                    cplx(1.0, 0.0);
            }
            else
            {
                Q[row][column] =
                    cplx(0.0, 0.0);
            }
        }
    }

    /*
     * Complex Givens rotations.
     */
QR_COLUMNS:
    for (int j = 0; j < N; j++)
    {
    QR_ROWS:
        for (int i = N - 1;
             i > j;
             i--)
        {
            double lower_power =
                complex_mag_squared(
                    R[i][j]
                );

            if (lower_power >
                EPS_SQUARED)
            {
                cplx a =
                    R[i - 1][j];

                cplx b =
                    R[i][j];

                double magnitude_a =
                    complex_magnitude(a);

                double rotation_norm =
                    std::sqrt(
                        magnitude_a *
                        magnitude_a +
                        complex_mag_squared(b)
                    );

                if (rotation_norm > EPS)
                {
                    double c =
                        magnitude_a /
                        rotation_norm;

                    cplx alpha;

                    if (magnitude_a < EPS)
                    {
                        alpha =
                            cplx(1.0, 0.0);
                    }
                    else
                    {
                        alpha =
                            complex_divide_real(
                                a,
                                magnitude_a
                            );
                    }

                    cplx s =
                        complex_divide_real(
                            complex_multiply(
                                alpha,
                                complex_conjugate(b)
                            ),
                            rotation_norm
                        );

                    cplx conjugate_s =
                        complex_conjugate(s);

                UPDATE_R:
                    for (int k = j;
                         k < N;
                         k++)
                    {
#pragma HLS PIPELINE

                        cplx upper_old =
                            R[i - 1][k];

                        cplx lower_old =
                            R[i][k];

                        cplx upper_new =
                            complex_add(
                                complex_scale(
                                    upper_old,
                                    c
                                ),
                                complex_multiply(
                                    s,
                                    lower_old
                                )
                            );

                        cplx lower_new =
                            complex_subtract(
                                complex_scale(
                                    lower_old,
                                    c
                                ),
                                complex_multiply(
                                    conjugate_s,
                                    upper_old
                                )
                            );

                        R[i - 1][k] =
                            upper_new;

                        R[i][k] =
                            lower_new;
                    }

                    R[i][j] =
                        cplx(0.0, 0.0);

                UPDATE_Q:
                    for (int k = 0;
                         k < N;
                         k++)
                    {
#pragma HLS PIPELINE

                        cplx left_old =
                            Q[k][i - 1];

                        cplx right_old =
                            Q[k][i];

                        cplx left_new =
                            complex_add(
                                complex_scale(
                                    left_old,
                                    c
                                ),
                                complex_multiply(
                                    conjugate_s,
                                    right_old
                                )
                            );

                        cplx right_new =
                            complex_subtract(
                                complex_scale(
                                    right_old,
                                    c
                                ),
                                complex_multiply(
                                    s,
                                    left_old
                                )
                            );

                        Q[k][i - 1] =
                            left_new;

                        Q[k][i] =
                            right_new;
                    }
                }
            }
        }
    }
}

This remains a complex Givens QR decomposition. It uses a complex s and its conjugate.

#include "qh_multiply.h"

void qh_multiply(
    const cplx Q[N][N],
    const cplx a[N],
    cplx y[N]
)
{
QH_OUTPUT_LOOP:
    for (int i = 0; i < N; i++)
    {
        double sum_real = 0.0;
        double sum_imag = 0.0;

    QH_ACCUMULATION_LOOP:
        for (int j = 0; j < N; j++)
        {
#pragma HLS PIPELINE

            /*
             * conj(Q[j][i]) * a[j]
             */
            double q_real =
                Q[j][i].real();

            double q_imag =
                Q[j][i].imag();

            double a_real =
                a[j].real();

            double a_imag =
                a[j].imag();

            sum_real +=
                q_real * a_real +
                q_imag * a_imag;

            sum_imag +=
                q_real * a_imag -
                q_imag * a_real;
        }

        y[i] =
            cplx(
                sum_real,
                sum_imag
            );
    }
}

backsub.cpp

#include "backsub.h"

static cplx complex_multiply(
    cplx first,
    cplx second
)
{
#pragma HLS INLINE

    double ar = first.real();
    double ai = first.imag();

    double br = second.real();
    double bi = second.imag();

    return cplx(
        ar * br - ai * bi,
        ar * bi + ai * br
    );
}

static cplx complex_subtract(
    cplx first,
    cplx second
)
{
#pragma HLS INLINE

    return cplx(
        first.real() - second.real(),
        first.imag() - second.imag()
    );
}

static cplx complex_divide(
    cplx numerator,
    cplx denominator
)
{
#pragma HLS INLINE

    double nr = numerator.real();
    double ni = numerator.imag();

    double dr = denominator.real();
    double di = denominator.imag();

    double denominator_power =
        dr * dr +
        di * di;

    /*
     * A valid Ru must not have a zero diagonal.
     * The testbench checks this before execution.
     */
    if (denominator_power < 1.0e-30)
    {
        return cplx(0.0, 0.0);
    }

    return cplx(
        (nr * dr + ni * di) /
            denominator_power,

        (ni * dr - nr * di) /
            denominator_power
    );
}

void back_substitution(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
BACKSUB_OUTER:
    for (int i = N - 1;
         i >= 0;
         i--)
    {
        cplx sum =
            b[i];

    BACKSUB_INNER:
        for (int j = i + 1;
             j < N;
             j++)
        {
#pragma HLS PIPELINE

            cplx product =
                complex_multiply(
                    R[i][j],
                    x[j]
                );

            sum =
                complex_subtract(
                    sum,
                    product
                );
        }

        x[i] =
            complex_divide(
                sum,
                R[i][i]
            );
    }
}

x[63] must be calculated before x[62]
x[62] must be calculated before x[61]
...

Therefore, some dependency warning is natural and should not be “fixed” by forcing II=1.

#include "normalize_weights.h"

/*
 * Manual complex division:
 *
 * numerator   = nr + j*ni
 * denominator = dr + j*di
 */
static cplx complex_divide(
    cplx numerator,
    cplx denominator
)
{
#pragma HLS INLINE

    double nr = numerator.real();
    double ni = numerator.imag();

    double dr = denominator.real();
    double di = denominator.imag();

    double denominator_power =
        dr * dr +
        di * di;

    /*
     * Prevent NaN for invalid input.
     *
     * In a valid MVDR calculation,
     * a^H * w_unnorm must not be zero.
     */
    if (denominator_power < 1.0e-30)
    {
        return cplx(0.0, 0.0);
    }

    double output_real =
        (
            nr * dr +
            ni * di
        ) /
        denominator_power;

    double output_imag =
        (
            ni * dr -
            nr * di
        ) /
        denominator_power;

    return cplx(
        output_real,
        output_imag
    );
}

void normalize_weights(
    const cplx a[N],
    const cplx w_in[N],
    cplx w_out[N]
)
{
    double denominator_real = 0.0;
    double denominator_imag = 0.0;

    /*
     * Calculate:
     *
     * denominator = a^H * w_in
     *
     * conj(a) * w:
     *
     * real = ar*wr + ai*wi
     * imag = ar*wi - ai*wr
     */
NORMALIZATION_DOT_PRODUCT:
    for (int i = 0; i < N; i++)
    {
#pragma HLS PIPELINE

        double ar =
            a[i].real();

        double ai =
            a[i].imag();

        double wr =
            w_in[i].real();

        double wi =
            w_in[i].imag();

        denominator_real +=
            ar * wr +
            ai * wi;

        denominator_imag +=
            ar * wi -
            ai * wr;
    }

    cplx denominator(
        denominator_real,
        denominator_imag
    );

    /*
     * Normalize every raw weight:
     *
     * w_out = w_in / denominator
     */
NORMALIZATION_OUTPUT_LOOP:
    for (int i = 0; i < N; i++)
    {
#pragma HLS PIPELINE

        w_out[i] =
            complex_divide(
                w_in[i],
                denominator
            );
    }
}

Because the first loop contains floating-point accumulation, the achieved II may be greater than one. The PIPELINE directive asks HLS to pipeline the loop, but data dependencies can make the achieved II larger than the requested or ideal value.

 *   Xtrain = 64 x 5000 complex training matrix
 *   az_deg = target azimuth
 *   el_deg = target elevation
 *   d      = spacing in wavelength units
 *
 * Output:
 *   w_out = 64 normalized MVDR weights
 */

    const cplx Xtrain[N][K],
    double az_deg,
    double el_deg,
    double d,
    cplx w_out[N]
);

---

mvdr_top.cpp

#include "mvdr_top.h"

#include "steering_vec.h"
#include "covariance.h"
#include "qr.h"
#include "qh_multiply.h"
#include "backsub.h"
#include "normalize_weights.h"

void mvdr_top(
    const cplx Xtrain[N][K],
    double az_deg,
    double el_deg,
    double d,
    cplx w_out[N]
)
{
    /*
     * Target steering vector.
     */
    cplx a_s[N];

    /*
     * Loaded covariance matrix.
     */
    cplx R[N][N];

    /*
     * QR-decomposition results.
     */
    cplx Q[N][N];
    cplx Ru[N][N];

    /*
     * Q^H * a_s.
     */
    cplx y[N];

    /*
     * Raw solution before MVDR normalization.
     */
    cplx w_unnorm[N];

    /*
     * Stage 1:
     *
     * Generate target steering vector.
     */
    steering_vec(
        az_deg,
        el_deg,
        d,
        a_s
    );

    /*
     * Stage 2:
     *
     * R = Xtrain * Xtrain^H / K
     *
     * Then diagonal loading is added.
     */
    covariance_matrix(
        Xtrain,
        R
    );

    /*
     * Stage 3:
     *
     * R = Q * Ru
     */
    givens_qr(
        R,
        Q,
        Ru
    );

    /*
     * Stage 4:
     *
     * y = Q^H * a_s
     */
    qh_multiply(
        Q,
        a_s,
        y
    );

    /*
     * Stage 5:
     *
     * Solve:
     *
     * Ru * w_unnorm = y
     */
    back_substitution(
        Ru,
        y,
        w_unnorm
    );

    /*
     * Stage 6:
     *
     * w_out =
     * w_unnorm / (a_s^H * w_unnorm)
     */
    normalize_weights(
        a_s,
        w_unnorm,
        w_out
    );
}

There are intentionally no additional pragmas inside this wrapper. The optimization pragmas already exist inside each module.

Do not add:

#pragma HLS DATAFLOW

yet. DATAFLOW is intended to overlap producers and consumers by creating channels between functions or loops. Your current stages exchange large complete matrices and use std::complex, so first obtain a stable combined synthesis before attempting task-level overlap.

4. One likely compilation issue

Your separate headers may all contain:

#define N 64
typedef std::complex<double> cplx;

Because the definitions are identical, it may compile normally. But if Vitis reports macro-redefinition messages, use this safer pattern in every header:

#ifndef MVDR_COMMON_DEFINITIONS
#define MVDR_COMMON_DEFINITIONS

#include <complex>

#define N 64
#define K 5000

typedef std::complex<double> cplx;

#endif

An even cleaner solution is one common header.

Optional mvdr_common.h

#ifndef MVDR_COMMON_H
#define MVDR_COMMON_H

#include <complex>

#define NX 8
#define NY 8
#define N  64
#define K  5000

#define PI 3.14159265358979323846

typedef std::complex<double> cplx;

#endif

Then every module header can begin with:

#include "mvdr_common.h"

However, do not refactor all headers immediately when your isolated projects are already working. First try the combined project using the current headers.

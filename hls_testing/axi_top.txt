/*
 * =====================================================
 * COVARIANCE FROM FLATTENED AXI ARRAYS
 * =====================================================
 *
 * Input arrangement:
 *
 * Xtrain_real[sensor * K + snapshot]
 * Xtrain_imag[sensor * K + snapshot]
 *
 * Covariance:
 *
 * R(i,j) = sum_k X(i,k) * conj(X(j,k)) / K
 *
 * Only the upper triangle is calculated.
 * The lower triangle is generated using:
 *
 * R(j,i) = conj(R(i,j))
 */
static void covariance_from_axi(
    const double Xtrain_real[MVDR_AXI_SAMPLES],
    const double Xtrain_imag[MVDR_AXI_SAMPLES],
    cplx R[N][N]
)
{
    const double INV_K =
        1.0 / (double)MVDR_AXI_K;

COVARIANCE_ROW:
    for (int i = 0; i < MVDR_AXI_N; i++)
    {
    COVARIANCE_COLUMN:
        for (int j = i; j < MVDR_AXI_N; j++)
        {
            double sum_real = 0.0;
            double sum_imag = 0.0;

        COVARIANCE_SNAPSHOT:
            for (
                int k = 0;
                k < MVDR_AXI_K;
                k++
            )
            {
                int index_i =
                    i * MVDR_AXI_K + k;

                int index_j =
                    j * MVDR_AXI_K + k;

                double xi_real =
                    Xtrain_real[index_i];

                double xi_imag =
                    Xtrain_imag[index_i];

                double xj_real =
                    Xtrain_real[index_j];

                double xj_imag =
                    Xtrain_imag[index_j];

                /*
                 * (xi_real + j xi_imag)
                 *
                 * multiplied by:
                 *
                 * conj(xj_real + j xj_imag)
                 */
                sum_real +=
                    xi_real * xj_real +
                    xi_imag * xj_imag;

                sum_imag +=
                    xi_imag * xj_real -
                    xi_real * xj_imag;
            }

            double covariance_real =
                sum_real * INV_K;

            double covariance_imag =
                sum_imag * INV_K;

            R[i][j] =
                cplx(
                    covariance_real,
                    covariance_imag
                );

            /*
             * Hermitian lower triangle.
             */
            if (i != j)
            {
                R[j][i] =
                    cplx(
                        covariance_real,
                        -covariance_imag
                    );
            }
        }
    }

    /*
     * Diagonal loading:
     *
     * delta = 0.01 * trace(R) / N
     */
    double trace_real = 0.0;

TRACE_LOOP:
    for (int i = 0; i < MVDR_AXI_N; i++)
    {
        trace_real +=
            R[i][i].real();
    }

    double delta =
        0.01 *
        trace_real /
        (double)MVDR_AXI_N;

DIAGONAL_LOADING_LOOP:
    for (int i = 0; i < MVDR_AXI_N; i++)
    {
        R[i][i] =
            cplx(
                R[i][i].real() + delta,
                0.0
            );
    }
}

/*
 * =====================================================
 * AXI TOP FUNCTION
 * =====================================================
 */

void mvdr_top_axi(
    const double Xtrain_real[MVDR_AXI_SAMPLES],
    const double Xtrain_imag[MVDR_AXI_SAMPLES],

    double az_deg,
    double el_deg,
    double d,

    double w_real[MVDR_AXI_N],
    double w_imag[MVDR_AXI_N]
)
{
    /*
     * One memory-master interface.
     *
     * This is simpler for the first Vivado
     * block-design integration.
     */
#pragma HLS INTERFACE mode=m_axi \
    port=Xtrain_real \
    offset=slave \
    bundle=gmem \
    depth=320000

#pragma HLS INTERFACE mode=m_axi \
    port=Xtrain_imag \
    offset=slave \
    bundle=gmem \
    depth=320000

#pragma HLS INTERFACE mode=m_axi \
    port=w_real \
    offset=slave \
    bundle=gmem \
    depth=64

#pragma HLS INTERFACE mode=m_axi \
    port=w_imag \
    offset=slave \
    bundle=gmem \
    depth=64

    /*
     * AXI-Lite control-register interface.
     *
     * The processor writes:
     * - DDR addresses
     * - target angles
     * - antenna spacing
     * - start command
     */
#pragma HLS INTERFACE mode=s_axilite \
    port=Xtrain_real \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=Xtrain_imag \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=w_real \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=w_imag \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=az_deg \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=el_deg \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=d \
    bundle=control

#pragma HLS INTERFACE mode=s_axilite \
    port=return \
    bundle=control

    /*
     * Internal matrices and vectors.
     */
    cplx a_s[N];

    cplx R[N][N];

    cplx Q[N][N];
    cplx Ru[N][N];

    cplx y[N];

    cplx w_unnorm[N];
    cplx w_complex[N];

    /*
     * Stage 1:
     * Generate target steering vector.
     */
    steering_vec(
        az_deg,
        el_deg,
        d,
        a_s
    );

    /*
     * Stage 2:
     * Read flattened data through AXI and
     * calculate loaded covariance matrix.
     */
    covariance_from_axi(
        Xtrain_real,
        Xtrain_imag,
        R
    );

    /*
     * Stage 3:
     * R = Q * Ru
     */
    givens_qr(
        R,
        Q,
        Ru
    );

    /*
     * Stage 4:
     * y = Q^H * a_s
     */
    qh_multiply(
        Q,
        a_s,
        y
    );

    /*
     * Stage 5:
     * Ru * w_unnorm = y
     */
    back_substitution(
        Ru,
        y,
        w_unnorm
    );

    /*
     * Stage 6:
     * Final distortionless normalization.
     */
    normalize_weights(
        a_s,
        w_unnorm,
        w_complex
    );

    /*
     * Stage 7:
     * Split complex output into separate
     * real and imaginary AXI buffers.
     */
WRITE_WEIGHTS:
    for (int i = 0; i < MVDR_AXI_N; i++)
    {
#pragma HLS PIPELINE

        w_real[i] =
            w_complex[i].real();

        w_imag[i] =
            w_complex[i].imag();
    }
}

/*
 * =====================================================
 * READ ALL NUMERICAL VALUES FROM A CSV FILE
 * =====================================================
 *
 * Supports:
 *   - comma-separated matrix files
 *   - one-value-per-line vector files
 *   - row-vector files
 *   - column-vector files
 */
static vector<double> read_csv_values(
    const char* filename
)
{
    ifstream file(filename);

    if (!file.is_open())
    {
        cerr
            << "ERROR: Cannot open file: "
            << filename
            << endl;

        exit(1);
    }

    vector<double> values;
    string line;

    while (getline(file, line))
    {
        /*
         * Replace commas with spaces so that
         * stringstream can read every number.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream line_stream(line);

        double value;

        while (line_stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*

static bool complex_is_finite(
    tb_cplx value
)
{
    return
        std::isfinite(value.real()) &&
        std::isfinite(value.imag());
}

/*
 * =====================================================
 * CHECK COMPLETE VECTOR FOR NaN OR INFINITY
 * =====================================================

)

}

/*
 * =====================================================
 * RELATIVE VECTOR ERROR
 * =====================================================
 *
 * ||w_hls - w_matlab||2
 * ---------------------
 *      ||w_matlab||2
 */
static double relative_vector_error(
    const tb_cplx reference[MVDR_AXI_N],
    const tb_cplx result[MVDR_AXI_N]
)
{
    double difference_power = 0.0;
    double reference_power = 0.0;

    for (
        int i = 0;
        i < MVDR_AXI_N;
        i++
    )
    {
        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double reference_real =
            reference[i].real();

        double reference_imag =
            reference[i].imag();

        difference_power +=
            difference_real *
            difference_real +

            difference_imag *
            difference_imag;

        reference_power +=
            reference_real *
            reference_real +

            reference_imag *
            reference_imag;
    }

    if (reference_power < 1.0e-30)
    {
        cerr
            << "ERROR: MATLAB weight vector "
            << "has zero norm."
            << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        difference_power /
        reference_power
    );
}

/*
 * =====================================================
 * MAXIMUM INDIVIDUAL WEIGHT ERROR
 * =====================================================
 */
static double maximum_element_error(
    const tb_cplx reference[MVDR_AXI_N],
    const tb_cplx result[MVDR_AXI_N]
)
{
    double maximum_error = 0.0;

    for (
        int i = 0;
        i < MVDR_AXI_N;
        i++
    )
    {
        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double element_error =
            std::sqrt(
                difference_real *
                difference_real +

                difference_imag *
                difference_imag
            );

        if (element_error > maximum_error)
        {
            maximum_error =
                element_error;
        }
    }

    return maximum_error;
}

/*
 * =====================================================
 * HERMITIAN DOT PRODUCT
 * =====================================================
 *
 * result = first^H * second
 *
 * Testbench only.
 */
static tb_cplx hermitian_dot_product(
    const tb_cplx first[MVDR_AXI_N],
    const tb_cplx second[MVDR_AXI_N]
)
{
    tb_cplx sum(
        0.0,
        0.0
    );

    for (
        int i = 0;
        i < MVDR_AXI_N;
        i++
    )
    {
        sum +=
            std::conj(first[i]) *
            second[i];
    }

    return sum;
}

/*
 * =====================================================
 * COMPLEX MAGNITUDE
 * =====================================================
 */
static double complex_magnitude(
    tb_cplx value
)
{
    return std::sqrt(
        value.real() *
        value.real() +

        value.imag() *
        value.imag()
    );
}

        Xtrain_imag[MVDR_AXI_SAMPLES];

    /*
     * AXI top outputs.
     */
    static double
        w_hls_real[MVDR_AXI_N];

    static double
        w_hls_imag[MVDR_AXI_N];

    /*
     * Complex vectors used only by the testbench.
     */
    static tb_cplx
        w_hls[MVDR_AXI_N];

    static tb_cplx
        w_matlab[MVDR_AXI_N];

    static tb_cplx
        a_s_matlab[MVDR_AXI_N];

    cout << endl;

    cout
        << "======================================"
        << endl;

    cout
        << "AXI-COMPATIBLE MVDR TOP TEST"
        << endl;

    cout
        << "======================================"
        << endl;

    /*
     * Load the complete 64 x 5000 training matrix
     * into flattened real and imaginary arrays.
     */
    load_flattened_xtrain(
        "Xtrain_real.csv",
        "Xtrain_imag.csv",

        Xtrain_real,
        Xtrain_imag
    );

    cout << endl;

    /*
     * Load MATLAB normalized MVDR weights.
     */
    load_complex_vector(
        "w_real.csv",
        "w_imag.csv",

        w_matlab
    );

    /*
     * Load MATLAB target steering vector.
     *
     * This is used only for final constraint and
     * target-gain verification.
     */
    load_complex_vector(
        "a_s_real.csv",
        "a_s_imag.csv",

        a_s_matlab
    );

    cout << endl;

    cout
        << "Running mvdr_top_axi..."
        << endl;

    /*
     * Execute the final AXI-compatible top.
     */
    mvdr_top_axi(
        Xtrain_real,
        Xtrain_imag,

        40.0,   // target azimuth in degrees
        0.0,    // target elevation in degrees
        0.5,    // element spacing in wavelengths

        w_hls_real,
        w_hls_imag
    );

    /*
     * Convert split AXI outputs back into complex
     * values for comparison with MATLAB.
     */
CONVERT_OUTPUT_LOOP:
    for (
        int i = 0;
        i < MVDR_AXI_N;
        i++
    )
    {
        w_hls[i] =
            tb_cplx(
                w_hls_real[i],
                w_hls_imag[i]
            );
    }

    /*
     * Detect NaN or infinity before calculating
     * any error values.
     */
    if (!vector_is_finite(w_hls))
    {
        cout << endl;

        cout
            << "AXI MVDR TOP TEST FAILED"
            << endl;

        return 1;
    }

    /*
     * Compare final HLS weights with MATLAB.
     */
    double relative_error =
        relative_vector_error(
            w_matlab,
            w_hls
        );

    double maximum_error =
        maximum_element_error(
            w_matlab,
            w_hls
        );

    /*
     * MVDR distortionless constraint:
     *
     * a_s^H * w = 1
     */
    tb_cplx distortionless_constraint =
        hermitian_dot_product(
            a_s_matlab,
            w_hls
        );

    double constraint_error =
        complex_magnitude(
            distortionless_constraint -
            tb_cplx(1.0, 0.0)
        );

    /*
     * Target response:
     *
     * |w^H * a_s| = 1
     */
    tb_cplx target_response =
        hermitian_dot_product(
            w_hls,
            a_s_matlab
        );

    double target_gain =
        complex_magnitude(
            target_response
        );

    double target_gain_error =
        std::fabs(
            target_gain - 1.0
        );

    cout << endl;

    cout
        << "===== AXI MVDR RESULTS ====="
        << endl;

    cout
        << "Relative weight error = "
        << relative_error
        << endl;

    cout
        << "Maximum element error = "
        << maximum_error
        << endl;

    cout
        << "a_s^H * w_hls         = "
        << distortionless_constraint
        << endl;

    cout
        << "Constraint error      = "
        << constraint_error
        << endl;

    cout
        << "|w_hls^H * a_s|       = "
        << target_gain
        << endl;

    cout
        << "Target gain error     = "
        << target_gain_error
        << endl;

    cout << endl;

    cout
        << "First five final MVDR weights:"
        << endl;

    for (
        int i = 0;
        i < 5;
        i++
    )
    {
        cout << endl;

        cout
            << "Weight "
            << i
            << endl;

        cout
            << "MATLAB = "
            << w_matlab[i]
            << endl;

        cout
            << "HLS    = "
            << w_hls[i]
            << endl;
    }

    /*
     * Initial double-precision tolerance.
     *
     * If the result is slightly above 1e-6 but
     * still very small, inspect the exact value
     * before changing this tolerance.
     */
    const double TOLERANCE =
        1.0e-6;

    bool passed =
        std::isfinite(relative_error) &&

        std::isfinite(maximum_error) &&

        std::isfinite(constraint_error) &&

        std::isfinite(target_gain_error) &&

        relative_error <
            TOLERANCE &&

        maximum_error <
            TOLERANCE &&

        constraint_error <
            TOLERANCE &&

        target_gain_error <
            TOLERANCE;

    cout << endl;

    if (passed)
    {
        cout
            << "======================================"
            << endl;

        cout
            << "AXI MVDR TOP TEST PASSED"
            << endl;

        cout
            << "======================================"
            << endl;

        return 0;
    }

    cout
        << "======================================"
        << endl;

    cout
        << "AXI MVDR TOP TEST FAILED"
        << endl;

    cout
        << "======================================"
        << endl;

    return 1;
}

Module 1: Back Substitution

Your current code:

void back_substitution(cplx R[][N], cplx b[], cplx x[])
{
    for (int i = N-1; i >= 0; i--) {
        x[i] = b[i];
        for (int j = i+1; j < N; j++)
            x[i] -= R[i][j] * x[j];
        x[i] /= R[i][i];
    }
}

is already almost HLS-compatible.

The only issue is:

typedef complex<double> cplx;

because Vitis HLS prefers:

#include <hls_complex.h>

typedef hls::complex<double> cplx;

---

HLS version of backsub.cpp

#include "backsub.h"

void back_substitution(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
#pragma HLS INLINE off

BACKSUB_OUTER:
    for(int i=N-1;i>=0;i--)
    {
        cplx sum = b[i];

BACKSUB_INNER:
        for(int j=i+1;j<N;j++)
        {
#pragma HLS PIPELINE II=1
            sum -= R[i][j]*x[j];
        }

        x[i] = sum/R[i][i];
    }
}

#include <iostream>
#include "backsub.h"

using namespace std;

int main()
{
    cplx R[N][N];
    cplx b[N];
    cplx x[N];

    for(int i=0;i<N;i++)
    {
        b[i]=1.0;

        for(int j=0;j<N;j++)
        {
            if(i==j)
                R[i][j]=1.0;
            else
                R[i][j]=0.0;
        }
    }

    back_substitution(R,b,x);

    for(int i=0;i<5;i++)
        cout<<x[i]<<endl;

    return 0;
}

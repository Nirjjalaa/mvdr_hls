Module 1: Back Substitution

Your current code:

void back_substitution(cplx R[][N], cplx b[], cplx x[])
{
    for (int i = N-1; i >= 0; i--) {
        x[i] = b[i];
        for (int j = i+1; j < N; j++)
            x[i] -= R[i][j] * x[j];
        x[i] /= R[i][i];
    }
}

is already almost HLS-compatible.

The only issue is:

typedef complex<double> cplx;

because Vitis HLS prefers:

#include <hls_complex.h>

typedef hls::complex<double> cplx;

HLS version of backsub.cpp

#include "backsub.h"

void back_substitution(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
#pragma HLS INLINE off

BACKSUB_OUTER:
    for(int i=N-1;i>=0;i--)
    {
        cplx sum = b[i];

BACKSUB_INNER:
        for(int j=i+1;j<N;j++)
        {
#pragma HLS PIPELINE II=1
            sum -= R[i][j]*x[j];
        }

        x[i] = sum/R[i][i];
    }
}

#include <iostream>
#include <fstream>
#include <cmath>
#include "backsub.h"

using namespace std;

void load_matrix(
    const char* realfile,
    const char* imagfile,
    cplx A[N][N]
)
{
    ifstream fr(realfile);
    ifstream fi(imagfile);

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            double re,im;
            fr >> re;
            fi >> im;
            A[i][j]=cplx(re,im);
        }
    }

    fr.close();
    fi.close();
}

void load_vector(
    const char* realfile,
    const char* imagfile,
    cplx x[N]
)
{
    ifstream fr(realfile);
    ifstream fi(imagfile);

    for(int i=0;i<N;i++)
    {
        double re,im;
        fr >> re;
        fi >> im;
        x[i]=cplx(re,im);
    }

    fr.close();
    fi.close();
}

double vector_error(
    cplx a[N],
    cplx b[N]
)
{
    double err=0;
    double norm=0;

    for(int i=0;i<N;i++)
    {
        err += std::norm(a[i]-b[i]);
        norm += std::norm(a[i]);
    }

    return sqrt(err)/sqrt(norm);
}

int main()
{
    cplx R[N][N];
    cplx y[N];
    cplx w_gold[N];
    cplx w_hls[N];

    load_matrix(
        "Ru_real.csv",
        "Ru_imag.csv",
        R
    );

    load_vector(
        "y_real.csv",
        "y_imag.csv",
        y
    );

    load_vector(
        "w_real.csv",
        "w_imag.csv",
        w_gold
    );

    back_substitution(
        R,
        y,
        w_hls
    );

    double err =
        vector_error(
            w_gold,
            w_hls
        );

    cout<<endl;
    cout<<"===== BACK SUBSTITUTION ====="<<endl;
    cout<<"Relative error = "<<err<<endl;

    cout<<endl;
    cout<<"First 5 weights:"<<endl;

    for(int i=0;i<5;i++)
    {
        cout
            <<i<<" : "
            <<w_gold[i]
            <<"   "
            <<w_hls[i]
            <<endl;
    }

    if(err<1e-6)
        cout<<"\nPASS"<<endl;
    else
        cout<<"\nFAIL"<<endl;

    return 0;
}

Step 6: What happens?

When you run:

Run C Simulation

Vitis copies all the testbench files and CSVs into its simulation directory automatically.

Then inside your testbench you can simply write:

ifstream f("Ru_real.csv");
ifstream f("y_real.csv");

without any path like:

"C:/mvdr_hls/verify_data/..."

because Vitis already placed them in the current simulation folder.


---

So the answer is:

❌ Don't hardcode paths like C:/mvdr_hls/...

❌ Don't create the CSV files inside VS Code

✅ Keep the CSV files in your project folder

✅ Add them under Testbench Files while creating the Vitis HLS project

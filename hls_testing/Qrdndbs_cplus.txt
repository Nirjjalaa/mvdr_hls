2. qr.cpp

#include "qr.h"
#include <cmath>

/*
 * Squared magnitude:
 *
 * |z|^2 = real(z)^2 + imag(z)^2
 *
 * This replaces std::norm(z).
 */
static double complex_mag_squared(cplx z)
{
    double real_part = z.real();
    double imag_part = z.imag();

    return real_part * real_part +
           imag_part * imag_part;
}

/*
 * Complex magnitude:
 *
 * |z| = sqrt(real(z)^2 + imag(z)^2)
 *
 * This replaces std::abs(z), which caused Vitis HLS
 * to call the unsupported cabs function.
 */
static double complex_magnitude(cplx z)
{
    double magnitude_squared =
        complex_mag_squared(z);

    return std::sqrt(magnitude_squared);
}

/*
 * Complex conjugate:
 *
 * conj(a + jb) = a - jb
 *
 * This replaces std::conj(z).
 */
static cplx complex_conjugate(cplx z)
{
    return cplx(
        z.real(),
        -z.imag()
    );
}

void givens_qr(
    cplx A[N][N],
    cplx Q[N][N],
    cplx R[N][N]
)
{
    const double EPS = 1.0e-12;
    const double EPS_SQUARED = 1.0e-24;

    /*
     * Initialize:
     *
     * R = A
     * Q = identity matrix
     */
INITIALIZE_ROWS:
    for (int row = 0; row < N; row++)
    {
    INITIALIZE_COLUMNS:
        for (int column = 0; column < N; column++)
        {
            R[row][column] =
                A[row][column];

            if (row == column)
            {
                Q[row][column] =
                    cplx(1.0, 0.0);
            }
            else
            {
                Q[row][column] =
                    cplx(0.0, 0.0);
            }
        }
    }

    /*
     * Complex Givens QR decomposition.
     *
     * This matches the MATLAB loop:
     *
     * for j = 1:n
     *     for i = m:-1:(j+1)
     */
QR_COLUMNS:
    for (int j = 0; j < N; j++)
    {
    QR_ROWS:
        for (int i = N - 1; i > j; i--)
        {
            /*
             * Check whether R(i,j) is already zero.
             *
             * MATLAB:
             * if abs(R(i,j)) > 1e-12
             */
            double lower_magnitude_squared =
                complex_mag_squared(R[i][j]);

            if (lower_magnitude_squared >
                EPS_SQUARED)
            {
                /*
                 * Adjacent matrix elements:
                 *
                 * a = R(i-1,j)
                 * b = R(i,j)
                 */
                cplx a = R[i - 1][j];
                cplx b = R[i][j];

                double magnitude_a =
                    complex_magnitude(a);

                double magnitude_b_squared =
                    complex_mag_squared(b);

                /*
                 * r = sqrt(|a|^2 + |b|^2)
                 */
                double rotation_norm =
                    std::sqrt(
                        magnitude_a * magnitude_a +
                        magnitude_b_squared
                    );

                if (rotation_norm > EPS)
                {
                    /*
                     * c = |a| / r
                     *
                     * c is real.
                     */
                    double c =
                        magnitude_a /
                        rotation_norm;

                    /*
                     * alpha = a / |a|
                     *
                     * When |a| is approximately zero,
                     * alpha is set to 1.
                     */
                    cplx alpha;

                    if (magnitude_a < EPS)
                    {
                        alpha =
                            cplx(1.0, 0.0);
                    }
                    else
                    {
                        alpha = cplx(
                            a.real() / magnitude_a,
                            a.imag() / magnitude_a
                        );
                    }

                    /*
                     * s = alpha * conj(b) / r
                     *
                     * s is generally complex.
                     */
                    cplx conjugate_b =
                        complex_conjugate(b);

                    cplx s =
                        (alpha * conjugate_b) /
                        rotation_norm;

                    cplx conjugate_s =
                        complex_conjugate(s);

                    /*
                     * Complex Givens rotation:
                     *
                     *     [ c          s ]
                     * G = [              ]
                     *     [ -conj(s)  c ]
                     *
                     * Update:
                     *
                     * R = G * R
                     */
                UPDATE_R:
                    for (int k = j; k < N; k++)
                    {
                        cplx upper_old =
                            R[i - 1][k];

                        cplx lower_old =
                            R[i][k];

                        cplx upper_new =
                            c * upper_old +
                            s * lower_old;

                        cplx lower_new =
                            -conjugate_s * upper_old +
                            c * lower_old;

                        R[i - 1][k] =
                            upper_new;

                        R[i][k] =
                            lower_new;
                    }

                    /*
                     * Force the eliminated element to zero.
                     *
                     * This avoids retaining a tiny floating-point
                     * residual below the diagonal.
                     */
                    R[i][j] =
                        cplx(0.0, 0.0);

                    /*
                     * MATLAB:
                     *
                     * Q(:,[i-1 i]) =
                     * Q(:,[i-1 i]) * G';
                     *
                     * For:
                     *
                     * G = [c, s; -conj(s), c]
                     *
                     * G' = [c, -s; conj(s), c]
                     */
                UPDATE_Q:
                    for (int k = 0; k < N; k++)
                    {
                        cplx left_old =
                            Q[k][i - 1];

                        cplx right_old =
                            Q[k][i];

                        cplx left_new =
                            c * left_old +
                            conjugate_s * right_old;

                        cplx right_new =
                            -s * left_old +
                            c * right_old;

                        Q[k][i - 1] =
                            left_new;

                        Q[k][i] =
                            right_new;
                    }
                }
            }
        }
    }
}

This is a complex Givens rotation because:

a and b are complex.

conj(s) is used.

Both real and imaginary parts of Q and R are updated.


The only ordinary mathematical library operation is std::sqrt(double). AMD documents synthesis support for float, double, and standard real-valued math functions such as sqrt() in Vitis HLS.

vector<double> read_csv_values(
    const char* filename
)
{
    ifstream file(filename);

    if (!file.is_open())
    {
        cerr << "ERROR: Cannot open file: "
             << filename << endl;

        exit(1);
    }

    vector<double> values;
    string line;

    while (getline(file, line))
    {
        /*
         * Convert commas to spaces.
         *
         * stringstream can then read every value
         * using the >> operator.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream stream(line);
        double value;

        while (stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*
 * Relative Frobenius error:
 *
 * sqrt(
 *   sum |result-reference|^2
 *   --------------------------------
 *   sum |reference|^2
 * )
 */
double relative_matrix_error(
    cplx reference[N][N],
    cplx result[N][N]
)
{
    double error_power = 0.0;
    double reference_power = 0.0;

    for (int row = 0; row < N; row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {
            double difference_real =
                result[row][column].real() -
                reference[row][column].real();

            double difference_imag =
                result[row][column].imag() -
                reference[row][column].imag();

            double reference_real =
                reference[row][column].real();

            double reference_imag =
                reference[row][column].imag();

            error_power +=
                difference_real *
                difference_real +
                difference_imag *
                difference_imag;

            reference_power +=
                reference_real *
                reference_real +
                reference_imag *
                reference_imag;
        }
    }

    if (reference_power < 1.0e-30)
    {
        cerr
            << "ERROR: Reference matrix norm is zero."
            << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        error_power /
        reference_power
    );
}

 */
double reconstruction_error(
    cplx input_R[N][N],
    cplx Q[N][N],
    cplx Ru[N][N]
)
{
    double error_power = 0.0;
    double input_power = 0.0;

    for (int row = 0; row < N; row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {

                cplx(0.0, 0.0);

                    Ru[k][column];

            cplx difference =
                reconstructed -
                input_R[row][column];

            error_power +=
                std::norm(difference);

            input_power +=
                std::norm(
                    input_R[row][column]
                );
        }
    }

    if (input_power < 1.0e-30)
    {
        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        error_power /
        input_power
    );
}

#include "backsub.h"

/*
 * Complex multiplication:
 *
 * (ar + j*ai)(br + j*bi)
 *
 * real = ar*br - ai*bi
 * imag = ar*bi + ai*br
 */
static cplx complex_multiply(
    cplx a,
    cplx b
)
{
    double ar = a.real();
    double ai = a.imag();

    double br = b.real();
    double bi = b.imag();

    double result_real =
        ar * br - ai * bi;

    double result_imag =
        ar * bi + ai * br;

    return cplx(
        result_real,
        result_imag
    );
}

/*
 * Complex subtraction:
 *
 * result = a - b
 */
static cplx complex_subtract(
    cplx a,
    cplx b
)
{
    return cplx(
        a.real() - b.real(),
        a.imag() - b.imag()
    );
}

/*
 * Complex division:
 *
 *      nr + j*ni
 * ------------------
 *      dr + j*di
 *
 * denominator = dr^2 + di^2
 *
 * real = (nr*dr + ni*di) / denominator
 * imag = (ni*dr - nr*di) / denominator
 */
static cplx complex_divide(
    cplx numerator,
    cplx denominator
)
{
    double nr = numerator.real();
    double ni = numerator.imag();

    double dr = denominator.real();
    double di = denominator.imag();

    double denominator_power =
        dr * dr + di * di;

    /*
     * Valid QRD output should not have a zero
     * diagonal. This guard prevents NaN if an
     * invalid matrix is supplied.
     */
    if (denominator_power < 1.0e-30)
    {
        return cplx(0.0, 0.0);
    }

    double result_real =
        (nr * dr + ni * di) /
        denominator_power;

    double result_imag =
        (ni * dr - nr * di) /
        denominator_power;

    return cplx(
        result_real,
        result_imag
    );
}

void back_substitution(
    cplx R[N][N],
    cplx b[N],
    cplx x[N]
)
{
BACKSUB_OUTER:
    for (int i = N - 1; i >= 0; i--)
    {
        cplx sum = b[i];

    BACKSUB_INNER:
        for (int j = i + 1; j < N; j++)
        {
            cplx product =
                complex_multiply(
                    R[i][j],
                    x[j]
                );

            sum =
                complex_subtract(
                    sum,
                    product
                );
        }

        x[i] =
            complex_divide(
                sum,
                R[i][i]
            );
    }
}

    while (getline(file, line))
    {
        /*
         * Replace commas with spaces.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream stream(line);
        double value;

        while (stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*
 * Load a 64 x 64 complex matrix.
 */
void load_complex_matrix(
    const char* real_filename,
    const char* imag_filename,
    cplx matrix[N][N]
)
{
    vector<double> real_values =
        read_csv_values(real_filename);

    vector<double> imag_values =
        read_csv_values(imag_filename);

    const int expected_values =
        N * N;

    cout << real_filename
         << " values found = "
         << real_values.size()
         << endl;

    cout << imag_filename
         << " values found = "
         << imag_values.size()
         << endl;

    if ((int)real_values.size() !=
        expected_values)
    {
        cerr << "ERROR: "
             << real_filename
             << " must contain "
             << expected_values
             << " values for a "
             << N << " x " << N
             << " matrix."
             << endl;

        exit(1);
    }

    if ((int)imag_values.size() !=
        expected_values)
    {
        cerr << "ERROR: "
             << imag_filename
             << " must contain "
             << expected_values
             << " values for a "
             << N << " x " << N
             << " matrix."
             << endl;

        exit(1);
    }

    int index = 0;

    for (int row = 0; row < N; row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {
            matrix[row][column] =
                cplx(
                    real_values[index],
                    imag_values[index]
                );

            index++;
        }
    }

    cout << "Loaded matrix successfully: "
         << N << " x " << N
         << endl;
}

/*
 * Checks whether a complex value is finite.
 */
bool complex_is_finite(cplx value)
{
    return
        std::isfinite(value.real()) &&
        std::isfinite(value.imag());
}

/*
 * Check every diagonal value of Ru.
 */
bool check_matrix_diagonal(
    cplx R[N][N]
)
{
    double minimum_diagonal_magnitude =
        numeric_limits<double>::infinity();

    int minimum_index = -1;

    for (int i = 0; i < N; i++)
    {
        if (!complex_is_finite(R[i][i]))
        {
            cerr << "ERROR: Ru diagonal contains "
                 << "NaN or infinity at index "
                 << i << endl;

            return false;
        }

        double real_part =
            R[i][i].real();

        double imag_part =
            R[i][i].imag();

        double magnitude =
            std::sqrt(
                real_part * real_part +
                imag_part * imag_part
            );

        if (magnitude <
            minimum_diagonal_magnitude)
        {
            minimum_diagonal_magnitude =
                magnitude;

            minimum_index = i;
        }

        if (magnitude < 1.0e-12)
        {
            cerr << "ERROR: Ru diagonal is zero "
                 << "or too small at index "
                 << i << endl;

            cerr << "Ru[" << i << "]["
                 << i << "] = "
                 << R[i][i]
                 << endl;

            return false;
        }
    }

    cout << "Smallest Ru diagonal magnitude = "
         << minimum_diagonal_magnitude
         << " at index "
         << minimum_index
         << endl;

    return true;
}

/*
 * Relative vector error:
 *
 * sqrt(
 *    sum |result-reference|^2
 *    ----------------------------
 *    sum |reference|^2
 * )
 */
double relative_vector_error(
    cplx reference[N],
    cplx result[N]
)
{
    double error_power = 0.0;
    double reference_power = 0.0;

    for (int i = 0; i < N; i++)
    {
        if (!complex_is_finite(result[i]))
        {
            cerr << "ERROR: HLS result contains "
                 << "NaN or infinity at index "
                 << i << endl;

            return
                numeric_limits<double>::infinity();
        }

        if (!complex_is_finite(reference[i]))
        {
            cerr << "ERROR: Golden result contains "
                 << "NaN or infinity at index "
                 << i << endl;

            return
                numeric_limits<double>::infinity();
        }

        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double reference_real =
            reference[i].real();

        double reference_imag =
            reference[i].imag();

        error_power +=
            difference_real * difference_real +
            difference_imag * difference_imag;

        reference_power +=
            reference_real * reference_real +
            reference_imag * reference_imag;
    }

    if (reference_power < 1.0e-30)
    {
        cerr << "ERROR: Golden-reference vector "
             << "has zero norm."
             << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        error_power /
        reference_power
    );
}

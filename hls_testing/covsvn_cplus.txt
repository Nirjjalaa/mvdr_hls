2. covariance.cpp

This is the synthesizable design file.

It avoids std::conj, std::abs, and std::norm in the design function, so it should not generate the cabs problem encountered in QRD.

#include "covariance.h"

void covariance_matrix(
    const cplx Xtrain[N][K],
    cplx R[N][N]
)
{
    const double inverse_K =
        1.0 / (double)K;

    /*
     * MATLAB equivalent:
     *
     * R = (Xtrain * Xtrain') / K;
     *
     * R(i,j) =
     * sum_k Xtrain(i,k) * conj(Xtrain(j,k)) / K
     */
COVARIANCE_ROW:
    for (int i = 0; i < N; i++)
    {
    COVARIANCE_COLUMN:
        for (int j = 0; j < N; j++)
        {
            double sum_real = 0.0;
            double sum_imag = 0.0;

        SNAPSHOT_LOOP:
            for (int k = 0; k < K; k++)
            {
                /*
                 * xi = ar + j*ai
                 * xj = br + j*bi
                 *
                 * xi * conj(xj)
                 *
                 * (ar + j*ai)(br - j*bi)
                 *
                 * Real:
                 * ar*br + ai*bi
                 *
                 * Imaginary:
                 * ai*br - ar*bi
                 */

                double ar =
                    Xtrain[i][k].real();

                double ai =
                    Xtrain[i][k].imag();

                double br =
                    Xtrain[j][k].real();

                double bi =
                    Xtrain[j][k].imag();

                sum_real +=
                    ar * br +
                    ai * bi;

                sum_imag +=
                    ai * br -
                    ar * bi;
            }

            R[i][j] = cplx(
                sum_real * inverse_K,
                sum_imag * inverse_K
            );
        }
    }

    /*
     * MATLAB equivalent:
     *
     * delta = 0.01 * trace(R) / N;
     */
    double trace_real = 0.0;

TRACE_LOOP:
    for (int i = 0; i < N; i++)
    {
        trace_real +=
            R[i][i].real();
    }

    double delta =
        0.01 *
        trace_real /
        (double)N;

    /*
     * MATLAB equivalent:
     *
     * R = R + delta * eye(N);
     */
DIAGONAL_LOADING_LOOP:
    for (int i = 0; i < N; i++)
    {
        double loaded_real =
            R[i][i].real() +
            delta;

        double existing_imag =
            R[i][i].imag();

        R[i][i] = cplx(
            loaded_real,
            existing_imag
        );
    }
}

/*
 * Read every number from a CSV file.
 *
 * MATLAB matrices contain:
 *
 * value,value,value,...
 *
 * This loader replaces commas with spaces,
 * allowing stringstream to read every number.
 */
vector<double> read_csv_values(
    const char* filename
)
{
    ifstream file(filename);

    if (!file.is_open())
    {
        cerr
            << "ERROR: Cannot open file: "
            << filename
            << endl;

        exit(1);
    }

    vector<double> values;
    string line;

    while (getline(file, line))
    {
        /*
         * Convert comma delimiters into spaces.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream line_stream(line);
        double value;

        while (line_stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*

bool complex_is_finite(
    cplx value
)
{
    return
        std::isfinite(value.real()) &&
        std::isfinite(value.imag());
}

/*
 * Check the entire matrix for NaN and infinity.
 */
bool matrix_is_finite(
    cplx matrix[N][N]
)
{
    for (int row = 0;
         row < N;
         row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {
            if (!complex_is_finite(
                    matrix[row][column]
                ))
            {
                cerr
                    << "ERROR: Matrix contains "
                    << "NaN or infinity at ["
                    << row << "]["
                    << column << "]"
                    << endl;

                cerr
                    << "Value = "
                    << matrix[row][column]
                    << endl;

                return false;
            }
        }
    }

    return true;
}

/*
 * Relative Frobenius matrix error:
 *
 * ||result - reference||F
 * ------------------------
 *      ||reference||F
 */
double relative_matrix_error(
    cplx reference[N][N],
    cplx result[N][N]
)
{
    double difference_power = 0.0;
    double reference_power = 0.0;

    for (int row = 0;
         row < N;
         row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {
            double difference_real =
                result[row][column].real() -
                reference[row][column].real();

            double difference_imag =
                result[row][column].imag() -
                reference[row][column].imag();

            double reference_real =
                reference[row][column].real();

            double reference_imag =
                reference[row][column].imag();

            difference_power +=
                difference_real *
                difference_real +
                difference_imag *
                difference_imag;

            reference_power +=
                reference_real *
                reference_real +
                reference_imag *
                reference_imag;
        }
    }

    if (reference_power <
        1.0e-30)
    {
        cerr
            << "ERROR: MATLAB reference R "
            << "has zero norm."
            << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        difference_power /
        reference_power
    );
}

/*
 * Maximum absolute complex-element error.
 */
double maximum_element_error(
    cplx reference[N][N],
    cplx result[N][N]
)
{
    double maximum_error = 0.0;

    for (int row = 0;
         row < N;
         row++)
    {
        for (int column = 0;
             column < N;
             column++)
        {
            double difference_real =
                result[row][column].real() -
                reference[row][column].real();

            double difference_imag =
                result[row][column].imag() -
                reference[row][column].imag();

            double error =
                std::sqrt(
                    difference_real *
                    difference_real +
                    difference_imag *
                    difference_imag
                );

            if (error > maximum_error)
            {
                maximum_error = error;
            }
        }
    }

    return maximum_error;
}

/*
 * Check Hermitian property:
 *
 * R(i,j) = conj(R(j,i))
 */
double hermitian_error(
    cplx matrix[N][N]
)
{
    double difference_power = 0.0;
    double matrix_power = 0.0;

    for (int i = 0;
         i < N;
         i++)
    {
        for (int j = 0;
             j < N;
             j++)
        {
            double conjugate_real =
                matrix[j][i].real();

            double conjugate_imag =
                -matrix[j][i].imag();

            double difference_real =
                matrix[i][j].real() -
                conjugate_real;

            double difference_imag =
                matrix[i][j].imag() -
                conjugate_imag;

            difference_power +=
                difference_real *
                difference_real +
                difference_imag *
                difference_imag;

            double matrix_real =
                matrix[i][j].real();

            double matrix_imag =
                matrix[i][j].imag();

            matrix_power +=
                matrix_real *
                matrix_real +
                matrix_imag *
                matrix_imag;
        }
    }

    if (matrix_power <
        1.0e-30)
    {
        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        difference_power /
        matrix_power
    );
}

/*
 * Calculate real trace.
 */
double real_trace(
    cplx matrix[N][N]
)
{
    double trace_value = 0.0;

    for (int i = 0;
         i < N;
         i++)
    {
        trace_value +=
            matrix[i][i].real();
    }

    return trace_value;
}

/*
 * Check that all diagonal elements are:
 *
 * 1. Positive
 * 2. Approximately real
 */
bool check_diagonal(
    cplx matrix[N][N]
)
{
    int diagonal_pass = 0;

    for (int i = 0;
         i < N;
         i++)
    {
        bool positive_real =
            matrix[i][i].real() > 0.0;

        bool imaginary_small =
            std::fabs(
                matrix[i][i].imag()
            ) < 1.0e-9;

        if (positive_real &&
            imaginary_small)
        {
            diagonal_pass++;
        }
        else
        {
            cerr
                << "Invalid diagonal at index "
                << i
                << ": "
                << matrix[i][i]
                << endl;
        }
    }

    cout
        << "Positive real diagonal = "
        << diagonal_pass
        << "/" << N
        << endl;

    return diagonal_pass == N;
}

2. steering_vec.cpp

This is the synthesizable design file.

#include "steering_vec.h"
#include <cmath>

void steering_vec(
    double az_deg,
    double el_deg,
    double d,
    cplx a[N]
)
{
    /*
     * Convert degrees into radians.
     */
    double az =
        az_deg * PI / 180.0;

    double el =
        el_deg * PI / 180.0;

    /*
     * Direction components used in your
     * MATLAB steering-vector function.
     */
    double u =
        std::sin(az) *
        std::cos(el);

    double v =
        std::sin(el);

    /*
     * Since N = 64:
     *
     * 1 / sqrt(64) = 0.125
     *
     * Using the exact value avoids an
     * unnecessary square-root unit.
     */
    const double normalization =
        0.125;

    int index = 0;

STEERING_Y:
    for (int y = 0; y < NY; y++)
    {
    STEERING_X:
        for (int x = 0; x < NX; x++)
        {
            /*
             * MATLAB equivalent:
             *
             * phase =
             * 2*pi*d*(x*u + y*v);
             */
            double phase =
                2.0 * PI * d *
                (
                    (double)x * u +
                    (double)y * v
                );

            /*
             * exp(j*phase) =
             * cos(phase) + j*sin(phase)
             */
            double output_real =
                normalization *
                std::cos(phase);

            double output_imag =
                normalization *
                std::sin(phase);

            a[index] = cplx(
                output_real,
                output_imag
            );

            index++;
        }
    }
}

This uses only real-valued sin() and cos() functions. It does not use complex abs, norm, or conj, so it will not generate the earlier cabs error. AMD’s HLS flow supports synthesizable mathematical functions for floating-point types. 

The array order exactly matches your MATLAB:

for y = 0:Ny-1
    for x = 0:Nx-1

/*
 * Relative vector error:
 *
 * ||result-reference||2
 * ---------------------
 *    ||reference||2
 */
double relative_vector_error(
    cplx reference[N],
    cplx result[N]
)
{
    double difference_power = 0.0;
    double reference_power = 0.0;

    for (int i = 0; i < N; i++)
    {
        if (!complex_is_finite(result[i]))
        {
            cerr
                << "ERROR: Result contains NaN "
                << "or infinity at index "
                << i
                << endl;

            return
                numeric_limits<double>::infinity();
        }

        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double reference_real =
            reference[i].real();

        double reference_imag =
            reference[i].imag();

        difference_power +=
            difference_real *
            difference_real +
            difference_imag *
            difference_imag;

        reference_power +=
            reference_real *
            reference_real +
            reference_imag *
            reference_imag;
    }

    if (reference_power < 1.0e-30)
    {
        cerr
            << "ERROR: Reference vector "
            << "has zero norm."
            << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        difference_power /
        reference_power
    );
}

/*
 * Maximum element error.
 */
double maximum_element_error(
    cplx reference[N],
    cplx result[N]
)
{
    double maximum_error = 0.0;

    for (int i = 0; i < N; i++)
    {
        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double element_error =
            std::sqrt(
                difference_real *
                difference_real +
                difference_imag *
                difference_imag
            );

        if (element_error > maximum_error)
        {
            maximum_error = element_error;
        }
    }

    return maximum_error;
}

/*
 * Check the vector norm.
 *
 * Every generated steering vector should have:
 *
 * ||a|| approximately equal to 1
 */
double vector_norm(
    cplx vector_input[N]
)
{
    double norm_power = 0.0;

    for (int i = 0; i < N; i++)
    {
        double real_part =
            vector_input[i].real();

        double imag_part =
            vector_input[i].imag();

        norm_power +=
            real_part * real_part +
            imag_part * imag_part;
    }

    return std::sqrt(norm_power);
}

This block computes:

y=Q^Ha_s

For each output element:

y_i=\sum_{j=0}^{63}Q_{j,i}^{*}a_{s,j}

It uses the complex conjugate transpose of Q, not an ordinary transpose. It does not perform back substitution or weight normalization.

2. qh_multiply.cpp

This is the synthesizable design file.

It manually performs conjugation and complex multiplication, avoiding std::conj, std::abs, and std::norm inside the hardware function.

#include "qh_multiply.h"

void qh_multiply(
    const cplx Q[N][N],
    const cplx a[N],
    cplx y[N]
)
{
QH_OUTPUT_LOOP:
    for (int i = 0; i < N; i++)
    {
        double sum_real = 0.0;
        double sum_imag = 0.0;

    QH_ACCUMULATION_LOOP:
        for (int j = 0; j < N; j++)
        {
            /*
             * Q^H(i,j) = conj(Q(j,i))
             *
             * Q(j,i) = qr + j*qi
             * conj(Q(j,i)) = qr - j*qi
             *
             * a(j) = ar + j*ai
             *
             * conj(Q(j,i)) * a(j):
             *
             * real = qr*ar + qi*ai
             * imag = qr*ai - qi*ar
             */

            double qr =
                Q[j][i].real();

            double qi =
                Q[j][i].imag();

            double ar =
                a[j].real();

            double ai =
                a[j].imag();

            double product_real =
                qr * ar +
                qi * ai;

            double product_imag =
                qr * ai -
                qi * ar;

            sum_real += product_real;
            sum_imag += product_imag;
        }

        y[i] = cplx(
            sum_real,
            sum_imag
        );
    }
}

/*
 * Reads every number from a CSV file.
 *
 * Supports:
 *   - comma-separated matrix rows
 *   - one-value-per-line vectors
 */
vector<double> read_csv_values(
    const char* filename
)
{
    ifstream file(filename);

    if (!file.is_open())
    {
        cerr
            << "ERROR: Cannot open file: "
            << filename
            << endl;

        exit(1);
    }

    vector<double> values;
    string line;

    while (getline(file, line))
    {
        /*
         * Convert commas into spaces.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream line_stream(line);
        double value;

        while (line_stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*

 *
 * y = Q^H * a
 *
 * This is not synthesized.

    cplx y_reference[N]

    for (int i = 0; i < N; i++)

After this, the next isolated block is weight normalization:

w_unnorm + a_s
        ↓
w = w_unnorm / (a_sᴴw_unnorm)












This block converts the raw back-substitution output into the final MVDR weights:

\text{denominator}=a_s^H w_{\text{unnorm}}

w_{\text{MVDR}}
=
\frac{w_{\text{unnorm}}}
{a_s^H w_{\text{unnorm}}}

After normalization:

a_s^H w_{\text{MVDR}}\approx1

and therefore:

\left|w_{\text{MVDR}}^H a_s\right|\approx1

The Euclidean norm norm(w) does not need to equal 1. It is the directional response  that is normalized.

AMD documents std::complex<T> as a supported representation for complex signals in Vitis HLS-related C++ flows. The testbench below is self-checking and returns zero only when all comparisons pass.

#include "normalize_weights.h"

/*
 * Manual complex division:
 *
 * numerator   = nr + j*ni
 * denominator = dr + j*di
 *
 * result real =
 * (nr*dr + ni*di) / (dr^2 + di^2)
 *
 * result imag =
 * (ni*dr - nr*di) / (dr^2 + di^2)
 */
static cplx complex_divide(
    cplx numerator,
    cplx denominator
)
{
    double nr = numerator.real();
    double ni = numerator.imag();

    double dr = denominator.real();
    double di = denominator.imag();

    double denominator_power =
        dr * dr + di * di;

    /*
     * A valid MVDR solution should not produce
     * a zero normalization factor.
     *
     * This guard prevents NaN if invalid input
     * data is supplied.
     */
    if (denominator_power < 1.0e-30)
    {
        return cplx(0.0, 0.0);
    }

    double output_real =
        (nr * dr + ni * di) /
        denominator_power;

    double output_imag =
        (ni * dr - nr * di) /
        denominator_power;

    return cplx(
        output_real,
        output_imag
    );
}

void normalize_weights(
    const cplx a[N],
    const cplx w_in[N],
    cplx w_out[N]
)
{
    double denominator_real = 0.0;
    double denominator_imag = 0.0;

    /*
     * Compute:
     *
     * denominator = a^H * w_in
     *
     * For each element:
     *
     * a = ar + j*ai
     * w = wr + j*wi
     *
     * conj(a) * w =
     *
     * real = ar*wr + ai*wi
     * imag = ar*wi - ai*wr
     */
NORMALIZATION_DOT_PRODUCT:
    for (int i = 0; i < N; i++)
    {
        double ar = a[i].real();
        double ai = a[i].imag();

        double wr = w_in[i].real();
        double wi = w_in[i].imag();

        denominator_real +=
            ar * wr +
            ai * wi;

        denominator_imag +=
            ar * wi -
            ai * wr;
    }

    cplx denominator(
        denominator_real,
        denominator_imag
    );

    /*
     * Divide every raw weight by the same
     * complex normalization factor.
     */
NORMALIZE_WEIGHT_LOOP:
    for (int i = 0; i < N; i++)
    {
        w_out[i] =
            complex_divide(
                w_in[i],
                denominator
            );
    }
}

/*
 * Read all numerical values from a CSV file.
 *
 * Handles:
 *   - one value per line
 *   - comma-separated values
 */
vector<double> read_csv_values(
    const char* filename
)
{
    ifstream file(filename);

    if (!file.is_open())
    {
        cerr
            << "ERROR: Cannot open file: "
            << filename
            << endl;

        exit(1);
    }

    vector<double> values;
    string line;

    while (getline(file, line))
    {
        /*
         * Replace commas with spaces.
         */
        for (size_t i = 0;
             i < line.size();
             i++)
        {
            if (line[i] == ',')
            {
                line[i] = ' ';
            }
        }

        stringstream line_stream(line);
        double value;

        while (line_stream >> value)
        {
            values.push_back(value);
        }
    }

    file.close();

    return values;
}

/*
 * Check whether one complex value is finite.
 */
bool complex_is_finite(
    cplx value
)
{
    return
        std::isfinite(value.real()) &&
        std::isfinite(value.imag());
}

/*
 * Check every vector value.
 */
bool vector_is_finite(
    cplx vector_input[N]
)
{
    for (int i = 0; i < N; i++)
    {
        if (!complex_is_finite(
                vector_input[i]
            ))
        {
            cerr
                << "ERROR: Non-finite value at index "
                << i
                << endl;

            cerr
                << "Value = "
                << vector_input[i]
                << endl;

            return false;
        }
    }

    return true;
}

/*
 * Relative vector error:
 *
 * ||result-reference||2
 * ---------------------
 *    ||reference||2
 */
double relative_vector_error(
    cplx reference[N],
    cplx result[N]
)
{
    double difference_power = 0.0;
    double reference_power = 0.0;

    for (int i = 0; i < N; i++)
    {
        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double reference_real =
            reference[i].real();

        double reference_imag =
            reference[i].imag();

        difference_power +=
            difference_real *
            difference_real +
            difference_imag *
            difference_imag;

        reference_power +=
            reference_real *
            reference_real +
            reference_imag *
            reference_imag;
    }

    if (reference_power < 1.0e-30)
    {
        cerr
            << "ERROR: MATLAB golden weight "
            << "vector has zero norm."
            << endl;

        return
            numeric_limits<double>::infinity();
    }

    return std::sqrt(
        difference_power /
        reference_power
    );
}

/*
 * Maximum individual complex-element error.
 */
double maximum_element_error(
    cplx reference[N],
    cplx result[N]
)
{
    double maximum_error = 0.0;

    for (int i = 0; i < N; i++)
    {
        double difference_real =
            result[i].real() -
            reference[i].real();

        double difference_imag =
            result[i].imag() -
            reference[i].imag();

        double element_error =
            std::sqrt(
                difference_real *
                difference_real +
                difference_imag *
                difference_imag
            );

        if (element_error > maximum_error)
        {
            maximum_error =
                element_error;
        }
    }

    return maximum_error;
}

/*
 * Calculate:
 *
 * result = a^H * b
 *
 * Testbench only.
 */
cplx hermitian_dot_product(
    cplx a[N],
    cplx b[N]
)
{
    cplx result =
        cplx(0.0, 0.0);

    for (int i = 0; i < N; i++)
    {
        result +=
            std::conj(a[i]) *
            b[i];
    }

    return result;
}

/*
 * Magnitude of a complex value.
 *
 * Testbench only.
 */
double complex_magnitude(
    cplx value
)
{
    double real_part =
        value.real();

    double imag_part =
        value.imag();

    return std::sqrt(
        real_part * real_part +
        imag_part * imag_part
    );
}

/*
 * Euclidean vector norm.
 *
 * This is printed only for information.
 * MVDR normalization does not force this norm to 1.
 */
double vector_norm(
    cplx vector_input[N]
)
{
    double power = 0.0;

    for (int i = 0; i < N; i++)
    {
        double real_part =
            vector_input[i].real();

        double imag_part =
            vector_input[i].imag();

        power +=
            real_part * real_part +
            imag_part * imag_part;
    }

    return std::sqrt(power);
}

qr.cpp

#include "qr.h"

void givens_qr(
    cplx A[N][N],
    cplx Q[N][N],
    cplx R[N][N]
)
{
    // Initialize
INIT:
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
#pragma HLS PIPELINE II=1
            R[i][j] = A[i][j];

            if(i==j)
                Q[i][j] = cplx(1.0,0.0);
            else
                Q[i][j] = cplx(0.0,0.0);
        }
    }

COL_LOOP:
    for(int j=0;j<N;j++)
    {
    ROW_LOOP:
        for(int i=N-1;i>=j+1;i--)
        {
            if(abs(R[i][j]) < 1e-12)
                continue;

            cplx a = R[i-1][j];
            cplx b = R[i][j];

            double r = sqrt(
                a.real()*a.real() +
                a.imag()*a.imag() +
                b.real()*b.real() +
                b.imag()*b.imag()
            );

            double c;
            cplx alpha;
            cplx s;

            if(abs(a) < 1e-12)
                alpha = cplx(1.0,0.0);
            else
                alpha = a/abs(a);

            c = abs(a)/r;
            s = alpha*conj(b)/r;

        APPLY_R:
            for(int k=j;k<N;k++)
            {
#pragma HLS PIPELINE II=1

                cplx tmp1 =
                    cplx(c,0)*R[i-1][k]
                    + s*R[i][k];

                cplx tmp2 =
                    -conj(s)*R[i-1][k]
                    + cplx(c,0)*R[i][k];

                R[i-1][k] = tmp1;
                R[i][k]   = tmp2;
            }

        APPLY_Q:
            for(int k=0;k<N;k++)
            {
#pragma HLS PIPELINE II=1

                cplx tmp1 =
                    Q[k][i-1]*cplx(c,0)
                    + Q[k][i]*conj(s);

                cplx tmp2 =
                    -Q[k][i-1]*s
                    + Q[k][i]*cplx(c,0);

                Q[k][i-1] = tmp1;
                Q[k][i]   = tmp2;
            }
        }
    }
}

Summary of isolated HLS verification order

Step	Top function	CSV files needed

1	back_substitution	Ru, y, w
2	givens_qr	R, Q, Ru
3	steering_vec	none or small golden values
4	covariance	Xtrain, R
5	mvdr_top	all required files


This is a good order because back substitution is the easiest kernel, QRD is the hardest kernel, and once QRD passes in HLS, the remaining MVDR pipeline is much more straightforward.

qrd_tb.cpp

#include <iostream>
#include <fstream>
#include <complex>
#include <cmath>

#define N 64

typedef std::complex<double> cplx;

void givens_qr(cplx A[][N], cplx Q[][N], cplx R[][N]);

// ===================================================
// CSV LOADER
// ===================================================
void load_csv(const char* fname, double M[N][N])
{
    std::ifstream f(fname);

    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
        {
            char comma;
            f >> M[i][j];
            if(f.peek()==',')
                f >> comma;
        }

    f.close();
}

// ===================================================
// TESTBENCH
// ===================================================
int main()
{
    double re[N][N];
    double im[N][N];

    cplx R_matlab[N][N];
    cplx Q_matlab[N][N];
    cplx Ru_matlab[N][N];

    // ==========================================
    // LOAD R
    // ==========================================
    load_csv("R_real.csv",re);
    load_csv("R_imag.csv",im);

    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
            R_matlab[i][j]=cplx(re[i][j],im[i][j]);

    // ==========================================
    // LOAD Q
    // ==========================================
    load_csv("Q_real.csv",re);
    load_csv("Q_imag.csv",im);

    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
            Q_matlab[i][j]=cplx(re[i][j],im[i][j]);

    // ==========================================
    // LOAD Ru
    // ==========================================
    load_csv("Ru_real.csv",re);
    load_csv("Ru_imag.csv",im);

    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
            Ru_matlab[i][j]=cplx(re[i][j],im[i][j]);

    // ==========================================
    // RUN HLS QRD
    // ==========================================
    cplx Q_cpp[N][N];
    cplx Ru_cpp[N][N];

    givens_qr(R_matlab,Q_cpp,Ru_cpp);

    std::cout
        << "\n===== HLS QR DECOMPOSITION =====\n";

    // ==========================================
    // CHECK 1
    // ==========================================
    double lower_err=0;

    for(int i=0;i<N;i++)
        for(int j=0;j<i;j++)
            lower_err+=std::abs(Ru_cpp[i][j]);

    lower_err/=(N*N);

    std::cout
        << "Lower triangular error : "
        << lower_err << "\n";

    // ==========================================
    // CHECK 2
    // ==========================================
    double orth_err=0;

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            cplx sum=0;

            for(int k=0;k<N;k++)
                sum+=std::conj(Q_cpp[k][i])
                     *Q_cpp[k][j];

            cplx expected=
                (i==j)?
                cplx(1,0):
                cplx(0,0);

            orth_err+=
                std::abs(sum-expected);
        }
    }

    orth_err/=(N*N);

    std::cout
        << "Orthogonality error    : "
        << orth_err << "\n";

    // ==========================================
    // CHECK 3
    // ==========================================
    double recon_err=0;
    double norm_R=0;

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            cplx sum=0;

            for(int k=0;k<N;k++)
                sum+=Q_cpp[i][k]
                    *Ru_cpp[k][j];

            recon_err+=
                std::norm(
                    sum-R_matlab[i][j]);

            norm_R+=
                std::norm(
                    R_matlab[i][j]);
        }
    }

    recon_err=
        sqrt(recon_err)
       /sqrt(norm_R);

    std::cout
        << "Reconstruction error  : "
        << recon_err << "\n";

    // ==========================================
    // CHECK 4
    // ==========================================
    double diag_err=0;

    for(int i=0;i<N;i++)
        diag_err+=
            std::abs(
                std::abs(
                    Ru_cpp[i][i])
                -
                std::abs(
                    Ru_matlab[i][i]));

    diag_err/=N;

    std::cout
        << "Diagonal match error  : "
        << diag_err << "\n";

    // ==========================================
    // PASS FAIL
    // ==========================================
    if(lower_err<1e-6 &&
       orth_err<1e-6 &&
       recon_err<1e-6)
    {
        std::cout
            << "\nQRD HLS TEST : PASS\n";
    }
    else
    {
        std::cout
            << "\nQRD HLS TEST : FAIL\n";
    }

    return 0;
}

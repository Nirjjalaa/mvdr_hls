#define COVARIANCE_H

#include <hls_complex.h>

#define N 64
#define K 5000

typedef hls::complex<double> cplx;

void covariance(
    cplx X[N][K],
    cplx R[N][N]
);

#endif


---

covariance.cpp

#include "covariance.h"

void covariance(
    cplx X[N][K],
    cplx R[N][N]
)
{
#pragma HLS INLINE off

    // initialize R
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
#pragma HLS PIPELINE
            R[i][j] = cplx(0,0);
        }
    }

    // compute covariance
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            cplx sum = cplx(0,0);

            for(int k=0;k<K;k++)
            {
#pragma HLS PIPELINE II=1
                sum += X[i][k] * hls::conj(X[j][k]);
            }

            R[i][j] = sum / (double)K;
        }
    }

    // diagonal loading
    double trace = 0.0;

    for(int i=0;i<N;i++)
        trace += R[i][i].real();

    double delta = 0.01 * trace / N;

    for(int i=0;i<N;i++)
        R[i][i] += cplx(delta,0);
}

qh_mult.cpp

#include "qh_mult.h"

void qh_mult(
    cplx Q[N][N],
    cplx a[N],
    cplx y[N]
)
{
#pragma HLS INLINE off

    for(int i = 0; i < N; i++)
    {
        cplx sum = cplx(0,0);

        for(int j = 0; j < N; j++)
        {
#pragma HLS PIPELINE II=1
            sum += hls::conj(Q[j][i]) * a[j];
        }

        y[i] = sum;
    }
}

steering_vec.cpp

#include "steering_vec.h"

void steering_vec(
    double az_deg,
    double el_deg,
    double d,
    cplx a[N]
)
{
#pragma HLS INTERFACE ap_ctrl_hs port=return

    double az = az_deg * PI / 180.0;
    double el = el_deg * PI / 180.0;

    double u = hls::sin(az) * hls::cos(el);
    double v = hls::sin(el);

    int k = 0;

    for(int y=0; y<NY; y++)
    {
        for(int x=0; x<NX; x++)
        {
#pragma HLS PIPELINE II=1

            double phase = 2.0*PI*d*(x*u + y*v);

            a[k] = cplx(
                hls::cos(phase),
                hls::sin(phase)
            );

            k++;
        }
    }

    double norm = hls::sqrt((double)N);

    for(int i=0;i<N;i++)
    {
#pragma HLS PIPELINE II=1
        a[i] /= norm;
    }
}

#define NORMALIZE_H

);

#endif


---

normalize.cpp

#include "normalize.h"

void normalize_weights(
    cplx w_in[N],
    cplx a[N],
    cplx w_out[N]
)
{
    cplx norm_factor = cplx(0.0,0.0);

    // Compute a^H * w
    for(int i=0;i<N;i++)
    {
#pragma HLS PIPELINE
        norm_factor += conj(a[i]) * w_in[i];
    }

    // Normalize
    for(int i=0;i<N;i++)
    {
#pragma HLS PIPELINE
        w_out[i] = w_in[i] / norm_factor;
    }
}

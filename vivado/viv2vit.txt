This code is a board smoke test. It generates deterministic training data inside the ARM processor, sends it to your MVDR IP through DDR, starts the IP, waits with timeout, then prints first weights.

This code assumes your HLS top is still the double version:

double Xtrain_imag[320000]
double w_real[64]
double w_imag[64]


---

main.c code for Vitis

#include "xparameters.h"
#include "xil_cache.h"
#include "xil_printf.h"
#include "xstatus.h"

#include "xmvdr_top_axi.h"

#include <stdint.h>
#include <math.h>

#define NX 8
#define NY 8
#define N  64
#define K  5000
#define SAMPLES (N*K)

#define PI 3.14159265358979323846

static double Xtrain_real[SAMPLES] __attribute__((aligned(64)));
static double Xtrain_imag[SAMPLES] __attribute__((aligned(64)));

static double w_real[N] __attribute__((aligned(64)));
static double w_imag[N] __attribute__((aligned(64)));

static double a_s_real[N];
static double a_s_imag[N];

static double a_i1_real[N];
static double a_i1_imag[N];

static double a_i2_real[N];
static double a_i2_imag[N];

static uint32_t rng_state = 123456789u;

static double rand_unit(void)
{
    rng_state = 1664525u * rng_state + 1013904223u;
    uint32_t v = (rng_state >> 8) & 0x00FFFFFFu;
    return ((double)v / 16777215.0) * 2.0 - 1.0;
}

static void make_steering(
    double az_deg,
    double el_deg,
    double d,
    double ar[N],
    double ai[N]
)
{
    double az = az_deg * PI / 180.0;
    double el = el_deg * PI / 180.0;

    double u = sin(az) * cos(el);
    double v = sin(el);

    int index = 0;

    for (int y = 0; y < NY; y++) {
        for (int x = 0; x < NX; x++) {
            double phase = 2.0 * PI * d * ((double)x * u + (double)y * v);

            ar[index] = cos(phase) / sqrt((double)N);
            ai[index] = sin(phase) / sqrt((double)N);

            index++;
        }
    }
}

static void generate_training_data(void)
{
    double d = 0.5;

    make_steering(40.0,   0.0,  d, a_s_real,  a_s_imag);
    make_steering(30.0,  10.0,  d, a_i1_real, a_i1_imag);
    make_steering(-60.0, -25.0, d, a_i2_real, a_i2_imag);

    for (int k = 0; k < K; k++) {
        double t = (double)k / (double)K;

        double sr = cos(2.0 * PI * 50.0 * t);
        double si = sin(2.0 * PI * 50.0 * t);

        double i1r = rand_unit();
        double i1i = rand_unit();

        double i2r = rand_unit();
        double i2i = rand_unit();

        for (int n = 0; n < N; n++) {
            double xr = 0.0;
            double xi = 0.0;

            /* target component: 10 * s * a_s */
            xr += 10.0 * (sr * a_s_real[n] - si * a_s_imag[n]);
            xi += 10.0 * (sr * a_s_imag[n] + si * a_s_real[n]);

            /* interference 1: 5 * i1 * a_i1 */
            xr += 5.0 * (i1r * a_i1_real[n] - i1i * a_i1_imag[n]);
            xi += 5.0 * (i1r * a_i1_imag[n] + i1i * a_i1_real[n]);

            /* interference 2: 5 * i2 * a_i2 */
            xr += 5.0 * (i2r * a_i2_real[n] - i2i * a_i2_imag[n]);
            xi += 5.0 * (i2r * a_i2_imag[n] + i2i * a_i2_real[n]);

            /* small noise */
            xr += 0.1 * rand_unit();
            xi += 0.1 * rand_unit();

            /*
             * IMPORTANT:
             * This assumes your HLS AXI top flattened the input as:
             * index = sensor_index * K + snapshot_index
             */
            int idx = n * K + k;

            Xtrain_real[idx] = xr;
            Xtrain_imag[idx] = xi;
        }
    }

    for (int n = 0; n < N; n++) {
        w_real[n] = 0.0;
        w_imag[n] = 0.0;
    }
}

static void print_double_6(const char *name, double value)
{
    int neg = 0;

    if (value < 0.0) {
        neg = 1;
        value = -value;
    }

    int scaled = (int)(value * 1000000.0 + 0.5);
    int whole = scaled / 1000000;
    int frac  = scaled % 1000000;

    xil_printf("%s", name);

    if (neg) {
        xil_printf("-");
    }

    xil_printf("%d.%06d", whole, frac);
}

int main(void)
{
    xil_printf("\r\n==============================\r\n");
    xil_printf("MVDR AXI IP Board Test Started\r\n");
    xil_printf("==============================\r\n");

    xil_printf("Generating training data...\r\n");
    generate_training_data();
    xil_printf("Training data generated.\r\n");

    xil_printf("First sample: ");
    print_double_6("", Xtrain_real[0]);
    xil_printf(" + j");
    print_double_6("", Xtrain_imag[0]);
    xil_printf("\r\n");

    XMvdr_top_axi ip;

    int status = XMvdr_top_axi_Initialize(&ip, XPAR_XMVDR_TOP_AXI_0_DEVICE_ID);

    if (status != XST_SUCCESS) {
        xil_printf("ERROR: MVDR IP initialization failed.\r\n");
        return -1;
    }

    xil_printf("MVDR IP initialized.\r\n");

    /*
     * Flush input and output buffers before IP starts.
     * This makes sure DDR has the latest CPU-written data.
     */
    Xil_DCacheFlushRange((UINTPTR)Xtrain_real, sizeof(Xtrain_real));
    Xil_DCacheFlushRange((UINTPTR)Xtrain_imag, sizeof(Xtrain_imag));
    Xil_DCacheFlushRange((UINTPTR)w_real, sizeof(w_real));
    Xil_DCacheFlushRange((UINTPTR)w_imag, sizeof(w_imag));

    /*
     * Set AXI pointer arguments.
     */
    XMvdr_top_axi_Set_Xtrain_real(&ip, (UINTPTR)Xtrain_real);
    XMvdr_top_axi_Set_Xtrain_imag(&ip, (UINTPTR)Xtrain_imag);

    XMvdr_top_axi_Set_w_real(&ip, (UINTPTR)w_real);
    XMvdr_top_axi_Set_w_imag(&ip, (UINTPTR)w_imag);

    /*
     * Set scalar arguments.
     */
    XMvdr_top_axi_Set_az_deg(&ip, 40.0);
    XMvdr_top_axi_Set_el_deg(&ip, 0.0);
    XMvdr_top_axi_Set_d(&ip, 0.5);

    xil_printf("Starting MVDR IP...\r\n");

    XMvdr_top_axi_Start(&ip);

    /*
     * Timeout wait.
     * Do not wait forever if something is wrong.
     */
    volatile unsigned int timeout = 0xFFFFFFFFu;

    while (!XMvdr_top_axi_IsDone(&ip) && timeout > 0u) {
        timeout--;
    }

    if (timeout == 0u) {
        xil_printf("ERROR: MVDR IP timeout. ap_done did not arrive.\r\n");
        xil_printf("Stop this run. Check AXI address, DDR path, cache, and clock/reset.\r\n");
        return -2;
    }

    xil_printf("MVDR IP done.\r\n");

    /*
     * Invalidate output buffers after IP finishes.
     * This makes CPU read the latest FPGA-written DDR data.
     */
    Xil_DCacheInvalidateRange((UINTPTR)w_real, sizeof(w_real));
    Xil_DCacheInvalidateRange((UINTPTR)w_imag, sizeof(w_imag));

    xil_printf("\r\nFirst 8 MVDR weights:\r\n");

    for (int i = 0; i < 8; i++) {
        xil_printf("w[%d] = ", i);
        print_double_6("", w_real[i]);
        xil_printf(" + j");
        print_double_6("", w_imag[i]);
        xil_printf("\r\n");
    }

    xil_printf("\r\nMVDR AXI IP Board Test Completed Successfully.\r\n");

    return 0;
}

  SAIF is just an activity report file.

Full form:

SAIF = Switching Activity Interchange Format

It tells Vivado:

Which signals toggled
How many times they toggled
How long they stayed 0 or 1
How active the clock/data paths really were

Vivado uses that to estimate power more accurately.

Why Vivado asks for SAIF

Without SAIF, Vivado guesses activity. It may assume something like:

many signals are switching regularly

So power can become pessimistic.

With SAIF, Vivado sees actual simulation activity:

This signal toggled often
This signal stayed idle
This DSP was active only during computation
This register hardly changed

Then the power report becomes more realistic.

Simple example

Without SAIF:

Vivado: I don't know activity, so I will estimate.
Power confidence: Low

With SAIF:

Vivado: I saw actual switching from simulation.
Power confidence: Medium/High

Full block design simulation with CIPS/NoC/DDR	Very difficult, may take 1–2 days or more

For your deadline, SAIF is not worth doing now.

Why I don’t recommend SAIF now

To generate SAIF properly, you need:

1. A proper RTL simulation testbench
2. Input stimulus for Xtrain_real/Xtrain_imag
3. Simulation long enough for MVDR IP to run
4. SAIF dump enabled
5. Import SAIF into implemented design
6. Rerun report_power

Your full MVDR design uses 320000 real samples + 320000 imaginary samples. RTL simulation with that much data can become slow.

So you may lose a full day just trying to get SAIF, and it will not help your main goal:

Run FPGA and print MVDR weights

What you should do instead

Since your implementation already passed timing and XSA is exported, do this:

1. Keep the current XSA safely.
2. Run a refined power report using realistic operating conditions.
3. Do Vitis board run.
4. Run MVDR IP once, print weights, stop.

In Vivado Tcl, you can do this quick power-report improvement:

open_run impl_1
set_operating_conditions -ambient_temp 25.0 -airflow 250 -heatsink medium
report_power -file mvdr_power_refined.rpt

This is much faster than SAIF.
